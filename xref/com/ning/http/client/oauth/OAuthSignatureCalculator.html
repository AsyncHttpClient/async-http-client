33  /**
34   * Simple OAuth signature calculator that can used for constructing client signatures
35   * for accessing services that use OAuth for authorization.
36   * <p/>
37   * Supports most common signature inclusion and calculation methods: HMAC-SHA1 for
38   * calculation, and Header inclusion as inclusion method. Nonce generation uses
39   * simple random numbers with base64 encoding.
40   *
41   * @author tatu (tatu.saloranta@iki.fi)
42   */
43  public class OAuthSignatureCalculator
44          implements SignatureCalculator {
45      public final static String HEADER_AUTHORIZATION = "Authorization";
46  
47      private static final String KEY_OAUTH_CONSUMER_KEY = "oauth_consumer_key";
48      private static final String KEY_OAUTH_NONCE = "oauth_nonce";
49      private static final String KEY_OAUTH_SIGNATURE = "oauth_signature";
50      private static final String KEY_OAUTH_SIGNATURE_METHOD = "oauth_signature_method";
51      private static final String KEY_OAUTH_TIMESTAMP = "oauth_timestamp";
52      private static final String KEY_OAUTH_TOKEN = "oauth_token";
53      private static final String KEY_OAUTH_VERSION = "oauth_version";
54  
55      private static final String OAUTH_VERSION_1_0 = "1.0";
56      private static final String OAUTH_SIGNATURE_METHOD = "HMAC-SHA1";
57  
58      /**
59       * To generate Nonce, need some (pseudo)randomness; no need for
60       * secure variant here.
61       */
62      protected final Random random;
63  
64      protected final byte[] nonceBuffer = new byte[16];
65  
66      protected final ThreadSafeHMAC mac;
67  
68      protected final ConsumerKey consumerAuth;
69  
70      protected final RequestToken userAuth;

72      /**
73       * @param consumerAuth Consumer key to use for signature calculation
74       * @param userAuth     Request/access token to use for signature calculation
75       */
76      public OAuthSignatureCalculator(ConsumerKey consumerAuth, RequestToken userAuth) {
77          mac = new ThreadSafeHMAC(consumerAuth, userAuth);
78          this.consumerAuth = consumerAuth;
79          this.userAuth = userAuth;
80          random = new Random(System.identityHashCode(this) + System.currentTimeMillis());
81      }

84  
85      public void calculateAndAddSignature(String baseURL, Request request, RequestBuilderBase<?> requestBuilder) {
86          String method = request.getMethod(); // POST etc
87          String nonce = generateNonce();
88          long timestamp = System.currentTimeMillis() / 1000L;
89          String signature = calculateSignature(method, baseURL, timestamp, nonce, request.getParams(), request.getQueryParams());
90          String headerValue = constructAuthHeader(signature, nonce, timestamp);
91          requestBuilder.setHeader(HEADER_AUTHORIZATION, headerValue);
92      }

94      /**
95       * Method for calculating OAuth signature using HMAC/SHA-1 method.
96       */
97      public String calculateSignature(String method, String baseURL, long oauthTimestamp, String nonce,
98                                       FluentStringsMap formParams, FluentStringsMap queryParams) {
99          StringBuilder signedText = new StringBuilder(100);
100         signedText.append(method); // POST / GET etc (nothing to URL encode)
101         signedText.append('&');
102 
103         /* 07-Oct-2010, tatu: URL may contain default port number; if so, need to extract
104          *   from base URL.
105          */
106         if (baseURL.startsWith("http:")) {
107             int i = baseURL.indexOf(":80/", 4);
108             if (i > 0) {
109                 baseURL = baseURL.substring(0, i) + baseURL.substring(i + 3);
110             }
111         } else if (baseURL.startsWith("https:")) {
112             int i = baseURL.indexOf(":443/", 5);
113             if (i > 0) {
114                 baseURL = baseURL.substring(0, i) + baseURL.substring(i + 4);
115             }
116         }
117         signedText.append(UTF8UrlEncoder.encode(baseURL));
118 
119         /**
120          * List of all query and form parameters added to this request; needed
121          * for calculating request signature
122          */
123         OAuthParameterSet allParameters = new OAuthParameterSet();
124 
125         // start with standard OAuth parameters we need
126         allParameters.add(KEY_OAUTH_CONSUMER_KEY, consumerAuth.getKey());
127         allParameters.add(KEY_OAUTH_NONCE, nonce);
128         allParameters.add(KEY_OAUTH_SIGNATURE_METHOD, OAUTH_SIGNATURE_METHOD);
129         allParameters.add(KEY_OAUTH_TIMESTAMP, String.valueOf(oauthTimestamp));
130         if (userAuth.getKey() != null) {
131             allParameters.add(KEY_OAUTH_TOKEN, userAuth.getKey());
132         }
133         allParameters.add(KEY_OAUTH_VERSION, OAUTH_VERSION_1_0);
134 
135         if (formParams != null) {
136             for (Map.Entry<String, List<String>> entry : formParams) {
137                 String key = entry.getKey();
138                 for (String value : entry.getValue()) {
139                     allParameters.add(key, value);
140                 }
141             }
142         }
143         if (queryParams != null) {
144             for (Map.Entry<String, List<String>> entry : queryParams) {
145                 String key = entry.getKey();
146                 for (String value : entry.getValue()) {
147                     allParameters.add(key, value);
148                 }
149             }
150         }
151         String encodedParams = allParameters.sortAndConcat();
152 
153         // and all that needs to be URL encoded (... again!)
154         signedText.append('&');
155         UTF8UrlEncoder.appendEncoded(signedText, encodedParams);
156 
157         byte[] rawBase = UTF8Codec.toUTF8(signedText.toString());
158         byte[] rawSignature = mac.digest(rawBase);
159         // and finally, base64 encoded... phew!
160         return Base64.encode(rawSignature);
161     }

163     /**
164      * Method used for constructing
165      */
166     public String constructAuthHeader(String signature, String nonce, long oauthTimestamp) {
167         StringBuilder sb = new StringBuilder(200);
168         sb.append("OAuth ");
169         sb.append(KEY_OAUTH_CONSUMER_KEY).append("=\"").append(consumerAuth.getKey()).append("\", ");
170         if (userAuth.getKey() != null) {
171             sb.append(KEY_OAUTH_TOKEN).append("=\"").append(userAuth.getKey()).append("\", ");
172         }
173         sb.append(KEY_OAUTH_SIGNATURE_METHOD).append("=\"").append(OAUTH_SIGNATURE_METHOD).append("\", ");
174 
175         // careful: base64 has chars that need URL encoding:
176         sb.append(KEY_OAUTH_SIGNATURE).append("=\"");
177         UTF8UrlEncoder.appendEncoded(sb, signature).append("\", ");
178         sb.append(KEY_OAUTH_TIMESTAMP).append("=\"").append(oauthTimestamp).append("\", ");
179 
180         // also: nonce may contain things that need URL encoding (esp. when using base64):
181         sb.append(KEY_OAUTH_NONCE).append("=\"");
182         UTF8UrlEncoder.appendEncoded(sb, nonce);
183         sb.append("\", ");
184 
185         sb.append(KEY_OAUTH_VERSION).append("=\"").append(OAUTH_VERSION_1_0).append("\"");
186         return sb.toString();
187     }

189     private synchronized String generateNonce() {
190         random.nextBytes(nonceBuffer);
191         // let's use base64 encoding over hex, slightly more compact than hex or decimals
192         return Base64.encode(nonceBuffer);
193 //      return String.valueOf(Math.abs(random.nextLong()));
194     }

196     /**
197      * Container for parameters used for calculating OAuth signature.
198      * About the only confusing aspect is that of whether entries are to be sorted
199      * before encoded or vice versa: if my reading is correct, encoding is to occur
200      * first, then sorting; although this should rarely matter (since sorting is primary
201      * by key, which usually has nothing to encode)... of course, rarely means that
202      * when it would occur it'd be harder to track down.
203      */
204     final static class OAuthParameterSet {
205         final private ArrayList<Parameter> allParameters = new ArrayList<Parameter>();
206 
207         public OAuthParameterSet() {
208         }
209 
210         public OAuthParameterSet add(String key, String value) {
211             Parameter p = new Parameter(UTF8UrlEncoder.encode(key), UTF8UrlEncoder.encode(value));
212             allParameters.add(p);
213             return this;
214         }
215 
216         public String sortAndConcat() {
217             // then sort them (AFTER encoding, important)
218             Parameter[] params = allParameters.toArray(new Parameter[allParameters.size()]);
219             Arrays.sort(params);
220 
221             // and build parameter section using pre-encoded pieces:
222             StringBuilder encodedParams = new StringBuilder(100);
223             for (Parameter param : params) {
224                 if (encodedParams.length() > 0) {
225                     encodedParams.append('&');
226                 }
227                 encodedParams.append(param.key()).append('=').append(param.value());
228             }
229             return encodedParams.toString();
230         }
231     }

233     /**
234      * Helper class for sorting query and form parameters that we need
235      */
236     final static class Parameter implements Comparable<Parameter> {
237         private final String key, value;
238 
239         public Parameter(String key, String value) {
240             this.key = key;
241             this.value = value;
242         }
243 
244         public String key() {
245             return key;
246         }
247 
248         public String value() {
249             return value;
250         }
251 
252         //@Override // silly 1.5; doesn't allow this for interfaces
253 
254         public int compareTo(Parameter other) {
255             int diff = key.compareTo(other.key);
256             if (diff == 0) {
257                 diff = value.compareTo(other.value);
258             }
259             return diff;
260         }
261 
262         @Override
263         public String toString() {
264             return key + "=" + value;
265         }
266 
267         @Override
268         public boolean equals(Object o) {
269             if (this == o) return true;
270             if (o == null || getClass() != o.getClass()) return false;
271 
272             Parameter parameter = (Parameter) o;
273 
274             if (!key.equals(parameter.key)) return false;
275             if (!value.equals(parameter.value)) return false;
276 
277             return true;
278         }
279 
280         @Override
281         public int hashCode() {
282             int result = key.hashCode();
283             result = 31 * result + value.hashCode();
284             return result;
285         }
286     }
