168         private URI toURI(boolean encode) {
169 
170             if (originalUri == null) {
171                 logger.debug("setUrl hasn't been invoked. Using http://localhost");
172                 originalUri = DEFAULT_REQUEST_URL;
173             }
174 
175             AsyncHttpProviderUtils.validateSupportedScheme(originalUri);
176 
177             StringBuilder builder = new StringBuilder();
178             builder.append(originalUri.getScheme()).append("://").append(originalUri.getRawAuthority());
179             if (isNonEmpty(originalUri.getRawPath())) {
180                 builder.append(originalUri.getRawPath());
181             } else {
182                 builder.append("/");
183             }
184 
185             if (isNonEmpty(queryParams)) {
186 
187                 builder.append("?");
188 
189                 for (Iterator<Entry<String, List<String>>> i = queryParams.iterator(); i.hasNext();) {
190                     Map.Entry<String, List<String>> param = i.next();
191                     String name = param.getKey();
192                     for (Iterator<String> j = param.getValue().iterator(); j.hasNext();) {
193                         String value = j.next();
194                         if (encode) {
195                             UTF8UrlEncoder.appendEncoded(builder, name);
196                         } else {
197                             builder.append(name);
198                         }
199                         if (value != null) {
200                             builder.append('=');
201                             if (encode) {
202                                 UTF8UrlEncoder.appendEncoded(builder, value);
203                             } else {
204                                 builder.append(value);
205                             }
206                         }
207                         if (j.hasNext()) {
208                             builder.append('&');
209                         }
210                     }
211                     if (i.hasNext()) {
212                         builder.append('&');
213                     }
214                 }
215             }
216 
217             return URI.create(builder.toString());
218         }

327 
328             sb.append("\t");
329             sb.append(method);
330             sb.append("\theaders:");
331             if (isNonEmpty(headers)) {

333                     sb.append("\t");

401     private void addQueryParameters(URI uri) {
402         if (isNonEmpty(uri.getRawQuery())) {
403             String[] queries = uri.getRawQuery().split("&");
404             int pos;
405             for (String query : queries) {
406                 pos = query.indexOf("=");
407                 if (pos <= 0) {
408                     addQueryParameter(query, null);
409                 } else {
410                     try {
411                         if (useRawUrl) {
412                             addQueryParameter(query.substring(0, pos), query.substring(pos + 1));
413                         } else {
414                             addQueryParameter(URLDecoder.decode(query.substring(0, pos), "UTF-8"), URLDecoder.decode(query.substring(pos + 1), "UTF-8"));
415                         }
416                     } catch (UnsupportedEncodingException e) {
417                         throw new RuntimeException(e);
418                     }
419                 }
420             }
421         }
422     }

631     public Request build() {
632 
633         /* Let's first calculate and inject signature, before finalizing actual build
634          * (order does not matter with current implementation but may in future)
635          */
636         if (signatureCalculator != null) {
637             String url = baseURL != null ? baseURL : request.originalUri.toString();
638             // Should not include query parameters, ensure:
639             int i = url.indexOf('?');
640             if (i != -1) {
641                 url = url.substring(0, i);
642             }
643             signatureCalculator.calculateAndAddSignature(url, request, this);
644         }
645 
646         try {
647             final String contentType = request.headers.getFirstValue("Content-Type");
648             if (contentType != null) {
649                 final String charset = AsyncHttpProviderUtils.parseCharset(contentType);
650                 if (charset != null) {
651                     // ensure that if charset is provided with the Content-Type header,
652                     // we propagate that down to the charset of the Request object
653                     request.charset = charset;
654                 }
655             }
656         } catch (Throwable e) {
657             // NoOp -- we can't fix the Content-Type or charset from here
658         }
659         if (request.length < 0 && request.streamData == null) {
660             // can't concatenate content-length
661             final String contentLength = request.headers.getFirstValue("Content-Length");
662 
663             if (contentLength != null) {
664                 try {
665                     request.length = Long.parseLong(contentLength);
666                 } catch (NumberFormatException e) {
667                     // NoOp -- we wdn't specify length so it will be chunked?
668                 }
669             }
670         }
671         return request;
672     }

674     public T addOrReplaceCookie(Cookie cookie) {
675         String cookieKey = cookie.getName();
676         boolean replace = false;
677         int index = 0;
678         for (Cookie c : request.cookies) {
679             if (c.getName().equals(cookieKey)) {
680                 replace = true;
681                 break;
682             }
683 
684             index++;
685         }
686         if (replace) {
687             ((ArrayList<Cookie>) request.cookies).set(index, cookie);
688         } else {
689             request.cookies.add(cookie);
690         }
691         return derived.cast(this);
