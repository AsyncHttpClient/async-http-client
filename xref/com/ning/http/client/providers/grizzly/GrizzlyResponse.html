65      public GrizzlyResponse(final HttpResponseStatus status,
66                             final HttpResponseHeaders headers,
67                             final List<HttpResponseBodyPart> bodyParts) {
68  
69          this.status = status;
70          this.headers = headers;
71          this.bodyParts = bodyParts;
72  
73          if (isNonEmpty(bodyParts)) {
74              if (bodyParts.size() == 1) {
75                  responseBody = ((GrizzlyResponseBodyPart) bodyParts.get(0)).getBodyBuffer();
76              } else {
77                  final Buffer firstBuffer = ((GrizzlyResponseBodyPart) bodyParts.get(0)).getBodyBuffer();
78                  final MemoryManager mm = MemoryManager.DEFAULT_MEMORY_MANAGER;
79                  Buffer constructedBodyBuffer = firstBuffer;
80                  for (int i = 1, len = bodyParts.size(); i < len; i++) {
81                      constructedBodyBuffer =
82                              Buffers.appendBuffers(mm,
83                                      constructedBodyBuffer,
84                                      ((GrizzlyResponseBodyPart) bodyParts.get(i)).getBodyBuffer());
85                  }
86                  responseBody = constructedBodyBuffer;
87              }
88          } else {
89              responseBody = Buffers.EMPTY_BUFFER;
90          }
91  
92      }

264     /**
265      * {@inheritDoc}
266      */
267     public List<Cookie> getCookies() {
268 
269         if (headers == null) {
270             return Collections.emptyList();
271         }
272 
273         if (cookies == null) {
274             List<String> values = headers.getHeaders().get("set-cookie");
275             if (isNonEmpty(values)) {
276                 CookiesBuilder.ServerCookiesBuilder builder =
277                     new CookiesBuilder.ServerCookiesBuilder(false, true);
278                 for (String header : values) {
279                     builder.parse(header);
280                 }
281                 cookies = convertCookies(builder.build());
282 
283             } else {
284                 cookies = Collections.emptyList();
285             }
286         }
287         return cookies;
288 
289     }

319     private List<Cookie> convertCookies(Cookies cookies) {
320 
321         final org.glassfish.grizzly.http.Cookie[] grizzlyCookies = cookies.get();
322         List<Cookie> convertedCookies = new ArrayList<Cookie>(grizzlyCookies.length);
323         for (org.glassfish.grizzly.http.Cookie gCookie : grizzlyCookies) {
324             convertedCookies.add(new Cookie(gCookie.getName(),
325                                    gCookie.getValue(),
326                                    gCookie.getValue(),
327                                    gCookie.getDomain(),
328                                    gCookie.getPath(),
329                                    -1L,
330                                    gCookie.getMaxAge(),
331                                    gCookie.isSecure(),
332                                    false));
333         }
334         return Collections.unmodifiableList(convertedCookies);
335 
336     }

339     private Charset getCharset(final String charset) {
340 
341         String charsetLocal = charset;
342 
343         if (charsetLocal == null) {
344             String contentType = getContentType();
345             if (contentType != null) {
346                 charsetLocal = AsyncHttpProviderUtils.parseCharset(contentType);
347             }
348         }
349 
350         if (charsetLocal == null) {
351             charsetLocal = Charsets.DEFAULT_CHARACTER_ENCODING;
352         }
353 
354         return Charsets.lookupCharset(charsetLocal);
355 
356     }
