16  package com.ning.http.client;
17  
18  /**
19   * Interface that allows injecting signature calculator into
20   * {@link RequestBuilder} so that signature calculation and inclusion can
21   * be added as a pluggable component.
22   *
23   * @since 1.1
24   */
25  public interface SignatureCalculator {
26      /**
27       * Method called when {@link RequestBuilder#build} method is called.
28       * Should first calculate signature information and then modify request
29       * (using passed {@link RequestBuilder}) to add signature (usually as
30       * an HTTP header).
31       *
32       * @param requestBuilder builder that can be used to modify request, usually
33       *                       by adding header that includes calculated signature. Be sure NOT to
34       *                       call {@link RequestBuilder#build} since this will cause infinite recursion
35       * @param request        Request that is being built; needed to access content to
36       *                       be signed
37       */
38      public void calculateAndAddSignature(String url, Request request,
39                                           RequestBuilderBase<?> requestBuilder);
40  }
