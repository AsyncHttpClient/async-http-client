69      public long read(ByteBuffer buffer) throws IOException {
70          try {
71              int overallLength = 0;
72  
73              final int maxLength = buffer.remaining();
74  
75              if (startPart == parts.size() && endWritten) {
76                  return -1;
77              }
78  
79              boolean full = false;
80              while (!full && !doneWritingParts) {
81                  com.ning.http.client.Part part = null;
82                  if (startPart < parts.size()) {
83                      part = parts.get(startPart);
84                  }
85                  if (currentFileChannel != null) {
86                      overallLength += currentFileChannel.read(buffer);
87  
88                      if (currentFileChannel.position() == currentFileChannel.size()) {
89                          currentFileChannel.close();
90                          currentFileChannel = null;
91                      }
92  
93                      if (overallLength == maxLength) {
94                          full = true;
95                      }
96                  } else if (currentStreamPosition > -1) {
97                      overallLength += writeToBuffer(buffer, maxLength - overallLength);
98  
99                      if (overallLength == maxLength) {
100                         full = true;
101                     }
102                     if (startPart == parts.size() && currentStream.available() == 0) {
103                         doneWritingParts = true;
104                     }
105                 } else if (part instanceof StringPart) {
106                     StringPart currentPart = (StringPart) part;
107 
108                     initializeStringPart(currentPart);
109 
110                     startPart++;
111                 } else if (part instanceof com.ning.http.client.StringPart) {
112                     StringPart currentPart = generateClientStringpart(part);
113 
114                     initializeStringPart(currentPart);
115 
116                     startPart++;
117                 } else if (part instanceof FilePart) {
118                     if (fileLocation == FileLocation.NONE) {
119                         currentFilePart = (FilePart) part;
120                         initializeFilePart(currentFilePart);
121                     } else if (fileLocation == FileLocation.START) {
122                         initializeFileBody(currentFilePart);
123                     } else if (fileLocation == FileLocation.MIDDLE) {
124                         initializeFileEnd(currentFilePart);
125                     } else if (fileLocation == FileLocation.END) {
126                         startPart++;
127                         fileLocation = FileLocation.NONE;
128                         if (startPart == parts.size() && currentStream.available() == 0) {
129                             doneWritingParts = true;
130                         }
131                     }
132                 } else if (part instanceof com.ning.http.client.FilePart) {
133                     if (fileLocation == FileLocation.NONE) {
134                         currentFilePart = generateClientFilePart(part);
135                         initializeFilePart(currentFilePart);
136                     } else if (fileLocation == FileLocation.START) {
137                         initializeFileBody(currentFilePart);
138                     } else if (fileLocation == FileLocation.MIDDLE) {
139                         initializeFileEnd(currentFilePart);
140                     } else if (fileLocation == FileLocation.END) {
141                         startPart++;
142                         fileLocation = FileLocation.NONE;
143                         if (startPart == parts.size() && currentStream.available() == 0) {
144                             doneWritingParts = true;
145                         }
146                     }
147                 } else if (part instanceof com.ning.http.client.ByteArrayPart) {
148                     com.ning.http.client.ByteArrayPart bytePart =
149                             (com.ning.http.client.ByteArrayPart) part;
150 
151                     if (fileLocation == FileLocation.NONE) {
152                         currentFilePart =
153                                 generateClientByteArrayPart(bytePart);
154 
155                         initializeFilePart(currentFilePart);
156                     } else if (fileLocation == FileLocation.START) {
157                         initializeByteArrayBody(currentFilePart);
158                     } else if (fileLocation == FileLocation.MIDDLE) {
159                         initializeFileEnd(currentFilePart);
160                     } else if (fileLocation == FileLocation.END) {
161                         startPart++;
162                         fileLocation = FileLocation.NONE;
163                         if (startPart == parts.size() && currentStream.available() == 0) {
164                             doneWritingParts = true;
165                         }
166                     }
167                 }
168             }
169 
170             if (doneWritingParts) {
171                 if (currentStreamPosition == -1) {
172                     ByteArrayOutputStream endWriter = new ByteArrayOutputStream();
173 
174                     Part.sendMessageEnd(endWriter, boundary);
175 
176                     initializeBuffer(endWriter.toByteArray());
177                 }
178 
179                 if (currentStreamPosition > -1) {
180                     overallLength += writeToBuffer(buffer, maxLength - overallLength);
181 
182                     if (currentStream.available() == 0) {
183                         currentStream.close();
184                         currentStreamPosition = -1;
185                         endWritten = true;
186                     }
187                 }
188             }
189             return overallLength;
190 
191         } catch (Exception e) {
192             logger.info("read exception", e);
193             return 0;
194         }
195     }

219     private void initializeFileBody(FilePart currentPart)
220             throws IOException {
221 
222         if (currentPart.getSource() instanceof FilePartSource) {
223 
224             FilePartSource source = (FilePartSource) currentPart.getSource();
225 
226             File file = source.getFile();
227 
228             RandomAccessFile raf = new RandomAccessFile(file, "r");
229             files.add(raf);
230 
231             currentFileChannel = raf.getChannel();
232 
233         } else {
234             // ByteArrayPartSource
235             PartSource partSource = currentPart.getSource();
236 
237             InputStream stream = partSource.createInputStream();
238 
239             byte[] bytes = new byte[(int) partSource.getLength()];
240 
241             stream.read(bytes);
242 
243             currentStream = new ByteArrayInputStream(bytes);
244 
245             currentStreamPosition = 0;
246         }
247 
248         fileLocation = FileLocation.MIDDLE;
249     }

271     private int writeToBuffer(ByteBuffer buffer, int length)
272             throws IOException {
273 
274         int available = currentStream.available();
275 
276         int writeLength = Math.min(available, length);
277 
278         byte[] bytes = new byte[writeLength];
279 
280         currentStream.read(bytes);
281 
282         buffer.put(bytes);
283 
284         if (available <= length) {
285             currentStream.close();
286             currentStreamPosition = -1;
287         } else {
288             currentStreamPosition += writeLength;
289         }
290 
291         return writeLength;
292     }

303     public long transferTo(long position, long count, WritableByteChannel target)
304             throws IOException {
305 
306         long overallLength = 0;
307 
308         if (startPart == parts.size()) {
309             return contentLength;
310         }
311 
312         int tempPart = startPart;
313 
314         for (com.ning.http.client.Part part : parts) {
315             if (part instanceof Part) {
316                 overallLength += handleMultiPart(target, (Part) part);
317             } else {
318                 overallLength += handleClientPart(target, part);
319             }
320 
321             tempPart++;
322         }
323         ByteArrayOutputStream endWriter = new ByteArrayOutputStream();
324 
325         Part.sendMessageEnd(endWriter, boundary);
326 
327         overallLength += writeToTarget(target, endWriter.toByteArray());
328 
329         startPart = tempPart;
330 
331         return overallLength;
332     }

334     private long handleClientPart(
335             WritableByteChannel target, com.ning.http.client.Part part) throws IOException {
336 
337         if (part.getClass().equals(com.ning.http.client.StringPart.class)) {
338             StringPart currentPart = generateClientStringpart(part);
339 
340             return handleStringPart(target, currentPart);
341         } else if (part.getClass().equals(com.ning.http.client.FilePart.class)) {
342             FilePart filePart = generateClientFilePart(part);
343 
344             return handleFilePart(target, filePart);
345         } else if (part.getClass().equals(com.ning.http.client.ByteArrayPart.class)) {
346             com.ning.http.client.ByteArrayPart bytePart = (com.ning.http.client.ByteArrayPart) part;
347 
348             FilePart filePart = generateClientByteArrayPart(bytePart);
349 
350             return handleByteArrayPart(target, filePart, bytePart.getData());
351         }
352 
353         return 0;
354     }

423     private long handleFilePart(WritableByteChannel target, FilePart filePart) throws IOException {
424     	FilePartStallHandler handler = new FilePartStallHandler(
425     		filePart.getStalledTime(), filePart);
426 
427     	handler.start();
428 
429         if (filePart.getSource() instanceof FilePartSource) {
430             int length = 0;
431 
432             length += handleFileHeaders(target, filePart);
433             FilePartSource source = (FilePartSource) filePart.getSource();
434 
435             File file = source.getFile();
436 
437             RandomAccessFile raf = new RandomAccessFile(file, "r");
438             files.add(raf);
439 
440             FileChannel fc = raf.getChannel();
441 
442             long l = file.length();
443             int fileLength = 0;
444             long nWrite = 0;
445             synchronized (fc) {
446                 while (fileLength != l) {
447                 	if(handler.isFailed()) {
448                 		logger.debug("Stalled error");
449                         throw new FileUploadStalledException();
450                 	}
451                     try {
452                         nWrite = fc.transferTo(fileLength, l, target);
453 
454                         if (nWrite == 0) {
455                             logger.info("Waiting for writing...");
456                             try {
457                                 fc.wait(50);
458                             } catch (InterruptedException e) {
459                                 logger.trace(e.getMessage(), e);
460                             }
461                         }
462                         else {
463                         	handler.writeHappened();
464                         }
465                     } catch (IOException ex) {
466                         String message = ex.getMessage();
467 
468                         // http://bugs.sun.com/view_bug.do?bug_id=5103988
469                         if (message != null && message.equalsIgnoreCase("Resource temporarily unavailable")) {
470                             try {
471                                 fc.wait(1000);
472                             } catch (InterruptedException e) {
473                                 logger.trace(e.getMessage(), e);
474                             }
475                             logger.warn("Experiencing NIO issue http://bugs.sun.com/view_bug.do?bug_id=5103988. Retrying");
476                             continue;
477                         } else {
478                             throw ex;
479                         }
480                     }
481                     fileLength += nWrite;
482                 }
483             }
484             handler.completed();
485 
486             fc.close();
487 
488             length += handleFileEnd(target, filePart);
489 
490             return length;
491         } else {
492             return handlePartSource(target, filePart);
493         }
494     }

496     private long handlePartSource(WritableByteChannel target, FilePart filePart) throws IOException {
497 
498         int length = 0;
499 
500         length += handleFileHeaders(target, filePart);
501 
502         PartSource partSource = filePart.getSource();
503 
504         InputStream stream = partSource.createInputStream();
505 
506         try {
507             int nRead = 0;
508             while (nRead != -1) {
509                 // Do not buffer the entire monster in memory.
510                 byte[] bytes = new byte[8192];
511                 nRead = stream.read(bytes);
512                 if (nRead > 0) {
513                     ByteArrayOutputStream bos = new ByteArrayOutputStream(nRead);
514                     bos.write(bytes, 0, nRead);
515                     length += writeToTarget(target, bos.toByteArray());
516                 }
517             }
518         } finally {
519             stream.close();
520         }
521         length += handleFileEnd(target, filePart);
522 
523         return length;
524     }

547     private long writeToTarget(WritableByteChannel target, byte[] bytes)
548             throws IOException {
549 
550         int written = 0;
551         int maxSpin = 0;
552         synchronized (bytes) {
553             ByteBuffer message = ByteBuffer.wrap(bytes);
554 
555             if (target instanceof SocketChannel) {
556                 final Selector selector = Selector.open();
557                 try {
558                     final SocketChannel channel = (SocketChannel) target;
559                     channel.register(selector, SelectionKey.OP_WRITE);
560 
561                     while (written < bytes.length) {
562                         selector.select(1000);
563                         maxSpin++;
564                         final Set<SelectionKey> selectedKeys = selector.selectedKeys();
565 
566                         for (SelectionKey key : selectedKeys) {
567                             if (key.isWritable()) {
568                                 written += target.write(message);
569                                 maxSpin = 0;
570                             }
571                         }
572 
573                         if (maxSpin >= 10) {
574                             throw new IOException("Unable to write on channel " + target);
575                         }
576                     }
577                 } finally {
578                     selector.close();
579                 }
580             } else {
581                 while ((target.isOpen()) && (written < bytes.length)) {
582                     long nWrite = target.write(message);
583                     written += nWrite;
584                     if (nWrite == 0 && maxSpin++ < 10) {
585                         logger.info("Waiting for writing...");
586                         try {
587                             bytes.wait(1000);
588                         } catch (InterruptedException e) {
589                             logger.trace(e.getMessage(), e);
590                         }
591                     } else {
592                         if (maxSpin >= 10) {
593                             throw new IOException("Unable to write on channel " + target);
594                         }
595                         maxSpin = 0;
596                     }
597                 }
598             }
599         }
600         return written;
601     }
