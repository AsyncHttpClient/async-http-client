164     public boolean cancel(boolean force) {
165         cancelTimeouts();
166 
167         if (isCancelled.getAndSet(true))
168             return false;
169 
170         try {
171             channel.getPipeline().getContext(NettyAsyncHttpProvider.class).setAttachment(new NettyAsyncHttpProvider.DiscardEvent());
172             channel.close();
173         } catch (Throwable t) {
174             // Ignore
175         }
176         if (!onThrowableCalled.getAndSet(true)) {
177             try {
178                 asyncHandler.onThrowable(new CancellationException());
179             } catch (Throwable t) {
180                 logger.warn("cancel", t);
181             }
182         }
183         latch.countDown();
184         runListeners();
185         return true;
186     }

237     /**
238      * {@inheritDoc}
239      */
240     /* @Override */
241     public V get(long l, TimeUnit tu) throws InterruptedException, TimeoutException, ExecutionException {
242         if (!isDone() && !isCancelled()) {
243             boolean expired = false;
244             if (l == -1) {
245                 latch.await();
246             } else {
247                 expired = !latch.await(l, tu);
248             }
249 
250             if (expired) {
251                 isCancelled.set(true);
252                 try {
253                     channel.getPipeline().getContext(NettyAsyncHttpProvider.class).setAttachment(new NettyAsyncHttpProvider.DiscardEvent());
254                     channel.close();
255                 } catch (Throwable t) {
256                     // Ignore
257                 }
258                 if (!onThrowableCalled.getAndSet(true)) {
259                     try {
260                         TimeoutException te = new TimeoutException(String.format("No response received after %s", l));
261                         try {
262                             asyncHandler.onThrowable(te);
263                         } catch (Throwable t) {
264                             logger.debug("asyncHandler.onThrowable", t);
265                         }
266                         throw new ExecutionException(te);
267                     } finally {
268                         cancelTimeouts();
269                     }
270                 }
271             }
272             isDone.set(true);
273 
274             ExecutionException e = exEx.getAndSet(null);
275             if (e != null) {
276                 throw e;
277             }
278         }
279         return getContent();
280     }

282     V getContent() throws ExecutionException {
283         ExecutionException e = exEx.getAndSet(null);
284         if (e != null) {
285             throw e;
286         }
287 
288         V update = content.get();
289         // No more retry
290         currentRetry.set(maxRetry);
291         if (exEx.get() == null && !contentProcessed.getAndSet(true)) {
292             try {
293                 update = asyncHandler.onCompleted();
294             } catch (Throwable ex) {
295                 if (!onThrowableCalled.getAndSet(true)) {
296                     try {
297                         try {
298                             asyncHandler.onThrowable(ex);
299                         } catch (Throwable t) {
300                             logger.debug("asyncHandler.onThrowable", t);
301                         }
302                         throw new RuntimeException(ex);
303                     } finally {
304                         cancelTimeouts();
305                     }
306                 }
307             }
308             content.compareAndSet(null, update);
309         }
310         return update;
311     }

313     public final void done() {
314         cancelTimeouts();
315 
316         try {
317             if (exEx.get() != null) {
318                 return;
319             }
320             getContent();
321             isDone.set(true);
322         } catch (ExecutionException t) {
323             return;
324         } catch (RuntimeException t) {
325         	Throwable exception = t.getCause() != null ? t.getCause() : t;
326         	exEx.compareAndSet(null, new ExecutionException(exception));
327 
328         } finally {
329             latch.countDown();
330         }
331 
332         runListeners();
333     }
334 
335     public final void abort(final Throwable t) {
336         cancelTimeouts();
337 
338         if (isDone.get() || isCancelled.getAndSet(true))
339             return;
340 
341         isCancelled.set(true);
342         exEx.compareAndSet(null, new ExecutionException(t));
343         if (onThrowableCalled.compareAndSet(false, true)) {
344             try {
345                 asyncHandler.onThrowable(t);
346             } catch (Throwable te) {
347                 logger.debug("asyncHandler.onThrowable", te);
348             }
349         }
350         latch.countDown();
351         runListeners();
352     }
