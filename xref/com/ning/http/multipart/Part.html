199     /**
200      * Write the content disposition header to the specified output stream
201      * 
202      * @param out The output stream
203      * @throws IOException If an IO problem occurs.
204      */
205     protected void sendDispositionHeader(OutputStream out) throws IOException {
206         out.write(CRLF_BYTES);
207         out.write(CONTENT_DISPOSITION_BYTES);
208         if (dispositionType != null)
209             out.write(MultipartEncodingUtil.getAsciiBytes(dispositionType));
210         else
211             out.write(FORM_DATA_DISPOSITION_TYPE_BYTES);
212 
213         if (getName() != null) {
214             out.write(NAME_BYTES);
215             out.write(QUOTE_BYTES);
216             out.write(MultipartEncodingUtil.getAsciiBytes(getName()));
217             out.write(QUOTE_BYTES);
218         }
219     }

431     /**
432      * Write all parts and the last boundary to the specified output stream.
433      * 
434      * @param out The stream to write to.
435      * @param parts The parts to write.
436      * @param partBoundary The ASCII bytes to use as the part boundary.
437      * @throws IOException If an I/O error occurs while writing the parts.
438      * @since 3.0
439      */
440     public static void sendParts(OutputStream out, Part[] parts, byte[] partBoundary) throws IOException {
441 
442         if (parts == null) {
443             throw new IllegalArgumentException("Parts may not be null");
444         }
445         if (partBoundary == null || partBoundary.length == 0) {
446             throw new IllegalArgumentException("partBoundary may not be empty");
447         }
448         for (Part part : parts) {
449             part.send(out, partBoundary);
450         }
451         out.write(EXTRA_BYTES);
452         out.write(partBoundary);
453         out.write(EXTRA_BYTES);
454         out.write(CRLF_BYTES);
455     }

486     /**
487      * Gets the length of the multipart message including the given parts.
488      * 
489      * @param parts The parts.
490      * @param partBoundary The ASCII bytes to use as the part boundary.
491      * @return The total length
492      * @throws IOException If an I/O error occurs while writing the parts.
493      * @since 3.0
494      */
495     public static long getLengthOfParts(Part[] parts, byte[] partBoundary) {
496 
497         try {
498             if (parts == null) {
499                 throw new IllegalArgumentException("Parts may not be null");
500             }
501             long total = 0;
502             for (Part part : parts) {
503                 long l = part.length(partBoundary);
504                 if (l < 0) {
505                     return -1;
506                 }
507                 total += l;
508             }
509             total += EXTRA_BYTES.length;
510             total += partBoundary.length;
511             total += EXTRA_BYTES.length;
512             total += CRLF_BYTES.length;
513             return total;
514         } catch (Exception e) {
515             LOGGER.error("An exception occurred while getting the length of the parts", e);
516             return 0L;
517         }
