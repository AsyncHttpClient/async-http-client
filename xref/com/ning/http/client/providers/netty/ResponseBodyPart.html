16  package com.ning.http.client.providers.netty;
17  
18  import com.ning.http.client.AsyncHttpProvider;
19  import com.ning.http.client.HttpResponseBodyPart;
20  import org.jboss.netty.buffer.ChannelBuffer;
21  import org.jboss.netty.handler.codec.http.HttpChunk;
22  import org.jboss.netty.handler.codec.http.HttpResponse;
23  
24  import java.io.IOException;
25  import java.io.OutputStream;
26  import java.net.URI;
27  import java.nio.ByteBuffer;
28  import java.util.concurrent.atomic.AtomicReference;
29  
30  /**
31   * A callback class used when an HTTP response body is received.
32   */
33  public class ResponseBodyPart extends HttpResponseBodyPart {
34  
35      private final ChannelBuffer content;
36      private final AtomicReference<byte[]> bytes = new AtomicReference<byte[]>(null);
37      private final boolean isLast;
38      private final int length;
39      private boolean closeConnection = false;
40  
41      public ResponseBodyPart(URI uri, HttpResponse response, AsyncHttpProvider provider, boolean last) {
42          this(uri, response, provider, null, last);
43      }
44  
45      public ResponseBodyPart(URI uri, HttpResponse response, AsyncHttpProvider provider, HttpChunk chunk, boolean last) {
46          super(uri, provider);
47          content = chunk != null ? chunk.getContent() : response.getContent();
48          length = content.readableBytes();
49          isLast = last;
50      }
51  
52      /**
53       * Return the response body's part bytes received.
54       *
55       * @return the response body's part bytes received.
56       */
57      public byte[] getBodyPartBytes() {
58  
59          if (bytes.get() != null) {
60              return bytes.get();
61          }
62  
63          byte[] b = ChannelBufferUtil.channelBuffer2bytes(content);
64          bytes.set(b);
65          return b;
66      }
67  
68      public int writeTo(OutputStream outputStream) throws IOException {
69          ChannelBuffer b = getChannelBuffer();
70          int read = b.readableBytes();
71          int index = b.readerIndex();
72          if (read > 0) {
73              b.readBytes(outputStream, read);
74          }
75          b.readerIndex(index);
76          return read;
77      }
78  
79      public ChannelBuffer getChannelBuffer() {
80          return content;
81      }
82  
83      @Override
84      public ByteBuffer getBodyByteBuffer() {
85          return ByteBuffer.wrap(getBodyPartBytes());
86      }
87  
88      /**
89       * {@inheritDoc}
90       */
91      @Override
92      public boolean isLast() {
93          return isLast;
94      }
95  
96      /**
97       * {@inheritDoc}
98       */
99      @Override
100     public void markUnderlyingConnectionAsClosed() {
101         closeConnection = true;
102     }
103 
104     /**
105      * {@inheritDoc}
106      */
107     @Override
108     public boolean closeUnderlyingConnection() {
109         return closeConnection;
110     }
111 
112     @Override
113     public int length() {
114         return length;
115     }
116 }
