33  /**
34   * An {@link AsyncHandler} which support resumable download, e.g when used with an {@link ResumableIOExceptionFilter},
35   * this handler can resume the download operation at the point it was before the interruption occured. This prevent having to
36   * download the entire file again. It's the responsibility of the {@link com.ning.http.client.listener.TransferListener}
37   * to track how many bytes has been transferred and to properly adjust the file's write position.
38   * <p/>
39   * In case of a JVM crash/shutdown, you can create an instance of this class and pass the last valid bytes position.
40   */
41  public class ResumableAsyncHandler<T> implements AsyncHandler<T> {
42      private final static Logger logger = LoggerFactory.getLogger(TransferCompletionHandler.class);
43      private final AtomicLong byteTransferred;
44      private String url;
45      private final ResumableProcessor resumableProcessor;
46      private final AsyncHandler<T> decoratedAsyncHandler;
47      private static Map<String, Long> resumableIndex;
48      private final static ResumableIndexThread resumeIndexThread = new ResumableIndexThread();
49      private ResponseBuilder responseBuilder = new ResponseBuilder();
50      private final boolean accumulateBody;
51      private ResumableListener resumableListener = new NULLResumableListener();
52  
53      private ResumableAsyncHandler(long byteTransferred,
54                                    ResumableProcessor resumableProcessor,
55                                    AsyncHandler<T> decoratedAsyncHandler,
56                                    boolean accumulateBody) {
57  
58          this.byteTransferred = new AtomicLong(byteTransferred);
59  
60          if (resumableProcessor == null) {
61              resumableProcessor = new NULLResumableHandler();
62          }
63          this.resumableProcessor = resumableProcessor;
64  
65          resumableIndex = resumableProcessor.load();
66          resumeIndexThread.addResumableProcessor(resumableProcessor);
67  
68          this.decoratedAsyncHandler = decoratedAsyncHandler;
69          this.accumulateBody = accumulateBody;
70      }

100     /**
101      * {@inheritDoc}
102      */
103     /* @Override */
104     public AsyncHandler.STATE onStatusReceived(final HttpResponseStatus status) throws Exception {
105         responseBuilder.accumulate(status);
106         if (status.getStatusCode() == 200 || status.getStatusCode() == 206) {
107             url = status.getUrl().toURL().toString();
108         } else {
109             return AsyncHandler.STATE.ABORT;
110         }
111 
112         if (decoratedAsyncHandler != null) {
113             return decoratedAsyncHandler.onStatusReceived(status);
114         }
115 
116         return AsyncHandler.STATE.CONTINUE;
117     }

119     /**
120      * {@inheritDoc}
121      */
122     /* @Override */
123     public void onThrowable(Throwable t) {
124         if (decoratedAsyncHandler != null) {
125             decoratedAsyncHandler.onThrowable(t);
126         } else {
127             logger.debug("", t);
128         }

131     /**
132      * {@inheritDoc}
133      */
134     /* @Override */
135     public AsyncHandler.STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception {
136 
137         if (accumulateBody) {
138             responseBuilder.accumulate(bodyPart);
139         }
140 
141         STATE state = STATE.CONTINUE;
142         try {
143             resumableListener.onBytesReceived(bodyPart.getBodyByteBuffer());
144         } catch (IOException ex) {
145             return AsyncHandler.STATE.ABORT;
146         }
147 
148         if (decoratedAsyncHandler != null) {
149             state = decoratedAsyncHandler.onBodyPartReceived(bodyPart);
150         }
151 
152         byteTransferred.addAndGet(bodyPart.getBodyPartBytes().length);
153         resumableProcessor.put(url, byteTransferred.get());
154 
155         return state;
156     }
157 
158     /**
159      * {@inheritDoc}
160      */
161     /* @Override */
162     public T onCompleted() throws Exception {
163         resumableProcessor.remove(url);
164         resumableListener.onAllBytesReceived();
165 
166         if (decoratedAsyncHandler != null) {
167             decoratedAsyncHandler.onCompleted();
168         }
169         // Not sure
170         return (T) responseBuilder.build();
171     }
172 
173     /**
174      * {@inheritDoc}
175      */
176     /* @Override */
177     public AsyncHandler.STATE onHeadersReceived(HttpResponseHeaders headers) throws Exception {
178         responseBuilder.accumulate(headers);
179         String contentLengthHeader = headers.getHeaders().getFirstValue("Content-Length");
180         if (contentLengthHeader != null) {
181             if (Long.parseLong(contentLengthHeader) == -1L) {
182                 return AsyncHandler.STATE.ABORT;
183             }
184         }
185 
186         if (decoratedAsyncHandler != null) {
187             return decoratedAsyncHandler.onHeadersReceived(headers);
188         }
189         return AsyncHandler.STATE.CONTINUE;
190     }

192     /**
193      * Invoke this API if you want to set the Range header on your {@link Request} based on the last valid bytes
194      * position.
195      *
196      * @param request {@link Request}
197      * @return a {@link Request} with the Range header properly set.
198      */
199     public Request adjustRequestRange(Request request) {
200 
201         if (resumableIndex.get(request.getUrl()) != null) {
202             byteTransferred.set(resumableIndex.get(request.getUrl()));
203         }
204 
205         // The Resumbale
206         if (resumableListener != null && resumableListener.length() > 0 && byteTransferred.get() != resumableListener.length()) {
207             byteTransferred.set(resumableListener.length());
208         }
209 
210         RequestBuilder builder = new RequestBuilder(request);
211         if (request.getHeaders().get("Range") == null && byteTransferred.get() != 0) {
212             builder.setHeader("Range", "bytes=" + byteTransferred.get() + "-");
213         }
214         return builder.build();
215     }

228     private static class ResumableIndexThread extends Thread {
229 
230         public final ConcurrentLinkedQueue<ResumableProcessor> resumableProcessors = new ConcurrentLinkedQueue<ResumableProcessor>();
231 
232         public ResumableIndexThread() {
233             Runtime.getRuntime().addShutdownHook(this);
234         }
235 
236         public void addResumableProcessor(ResumableProcessor p) {
237             resumableProcessors.offer(p);
238         }
239 
240         public void run() {
241             for (ResumableProcessor p : resumableProcessors) {
242                 p.save(resumableIndex);
243             }
244         }

247     /**
248      * An interface to implement in order to manage the way the incomplete file management are handled.
249      */
250     public static interface ResumableProcessor {
251 
252         /**
253          * Associate a key with the number of bytes sucessfully transferred.
254          *
255          * @param key              a key. The recommended way is to use an url.
256          * @param transferredBytes The number of bytes sucessfully transferred.
257          */
258         public void put(String key, long transferredBytes);
259 
260         /**
261          * Remove the key associate value.
262          *
263          * @param key key from which the value will be discarted
264          */
265         public void remove(String key);
266 
267         /**
268          * Save the current {@link Map} instance which contains information about the current transfer state.
269          * This method *only* invoked when the JVM is shutting down.
270          *
271          * @param map
272          */
273         public void save(Map<String, Long> map);
274 
275         /**
276          * Load the {@link Map} in memory, contains information about the transferred bytes.
277          *
278          * @return {@link Map}
279          */
280         public Map<String, Long> load();
281 
282     }
