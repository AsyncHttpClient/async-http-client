22  /**
23   * A {@link FilterContext} can be used to decorate {@link Request} and {@link AsyncHandler} from a list of {@link RequestFilter}.
24   * {@link RequestFilter} gets executed before the HTTP request is made to the remote server. Once the response bytes are
25   * received, a {@link FilterContext} is then passed to the list of {@link ResponseFilter}. {@link ResponseFilter}
26   * gets invoked before the response gets processed, e.g. before authorization, redirection and invokation of {@link AsyncHandler}
27   * gets processed.
28   * <p/>
29   * Invoking {@link com.ning.http.client.filter.FilterContext#getResponseStatus()} returns an instance of {@link HttpResponseStatus}
30   * that can be used to decide if the response processing should continue or not. You can stop the current response processing
31   * and replay the request but creating a {@link FilterContext}. The {@link com.ning.http.client.AsyncHttpProvider}
32   * will interrupt the processing and "replay" the associated {@link Request} instance.
33   */
34  public class FilterContext<T> {
35  
36      private final FilterContextBuilder b;
37  
38      /**
39       * Create a new {@link FilterContext}
40       *
41       * @param b a {@link FilterContextBuilder}
42       */
43      private FilterContext(FilterContextBuilder b) {
44          this.b = b;
45      }
46  
47      /**
48       * Return the original or decorated {@link AsyncHandler}
49       *
50       * @return the original or decorated {@link AsyncHandler}
51       */
52      public AsyncHandler<T> getAsyncHandler() {
53          return b.asyncHandler;
54      }
55  
56      /**
57       * Return the original or decorated {@link Request}
58       *
59       * @return the original or decorated {@link Request}
60       */
61      public Request getRequest() {
62          return b.request;
63      }
64  
65      /**
66       * Return the unprocessed response's {@link HttpResponseStatus}
67       *
68       * @return the unprocessed response's {@link HttpResponseStatus}
69       */
70      public HttpResponseStatus getResponseStatus() {
71          return b.responseStatus;
72      }
73  
74      /**
75       * Return the response {@link HttpResponseHeaders}
76       */
77      public HttpResponseHeaders getResponseHeaders() {
78          return b.headers;
79      }
80  
81      /**
82       * Return true if the current response's processing needs to be interrupted and a new {@link Request} be executed.
83       *
84       * @return true if the current response's processing needs to be interrupted and a new {@link Request} be executed.
85       */
86      public boolean replayRequest() {
87          return b.replayRequest;
88      }
89  
90      /**
91       * Return the {@link IOException}
92       *
93       * @return the {@link IOException}
94       */
95      public IOException getIOException() {
96          return b.ioException;
97      }
98  
99      public static class FilterContextBuilder<T> {
100         private AsyncHandler<T> asyncHandler = null;
101         private Request request = null;
102         private HttpResponseStatus responseStatus = null;
103         private boolean replayRequest = false;
104         private IOException ioException = null;
105         private HttpResponseHeaders headers;
106 
107         public FilterContextBuilder() {
108         }
109 
110         public FilterContextBuilder(FilterContext clone) {
111             asyncHandler = clone.getAsyncHandler();
112             request = clone.getRequest();
113             responseStatus = clone.getResponseStatus();
114             replayRequest = clone.replayRequest();
115             ioException = clone.getIOException();
116         }
117 
118         public AsyncHandler<T> getAsyncHandler() {
119             return asyncHandler;
120         }
121 
122         public FilterContextBuilder asyncHandler(AsyncHandler<T> asyncHandler) {
123             this.asyncHandler = asyncHandler;
124             return this;
125         }
126 
127         public Request getRequest() {
128             return request;
129         }
130 
131         public FilterContextBuilder request(Request request) {
132             this.request = request;
133             return this;
134         }
135 
136         public FilterContextBuilder responseStatus(HttpResponseStatus responseStatus) {
137             this.responseStatus = responseStatus;
138             return this;
139         }
140 
141         public FilterContextBuilder responseHeaders(HttpResponseHeaders headers) {
142             this.headers = headers;
143             return this;
144         }
145 
146         public FilterContextBuilder replayRequest(boolean replayRequest) {
147             this.replayRequest = replayRequest;
148             return this;
149         }
150 
151         public FilterContextBuilder ioException(IOException ioException) {
152             this.ioException = ioException;
153             return this;
154         }
155 
156         public FilterContext build() {
157             return new FilterContext(this);
158         }
159     }
