17  package com.ning.http.client;
18  
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  /**
23   * An {@link AsyncHandler} augmented with an {@link #onCompleted(Response)} convenience method which gets called
24   * when the {@link Response} processing is finished.  This class also implement the {@link ProgressAsyncHandler} callback,
25   * all doing nothing except returning {@link com.ning.http.client.AsyncHandler.STATE#CONTINUE}
26   *
27   * @param <T> Type of the value that will be returned by the associated {@link java.util.concurrent.Future}
28   */
29  public abstract class AsyncCompletionHandler<T> implements AsyncHandler<T>, ProgressAsyncHandler<T> {
30  
31      private final Logger log = LoggerFactory.getLogger(AsyncCompletionHandlerBase.class);
32      private final Response.ResponseBuilder builder = new Response.ResponseBuilder();
33  
34      /**
35       * {@inheritDoc}
36       */
37      public STATE onBodyPartReceived(final HttpResponseBodyPart content) throws Exception {
38          builder.accumulate(content);
39          return STATE.CONTINUE;
40      }
41  
42      /**
43       * {@inheritDoc}
44       */
45      public STATE onStatusReceived(final HttpResponseStatus status) throws Exception {
46          builder.reset();
47          builder.accumulate(status);
48          return STATE.CONTINUE;
49      }
50  
51      /**
52       * {@inheritDoc}
53       */
54      public STATE onHeadersReceived(final HttpResponseHeaders headers) throws Exception {
55          builder.accumulate(headers);
56          return STATE.CONTINUE;
57      }
58  
59      /**
60       * {@inheritDoc}
61       */
62      public final T onCompleted() throws Exception {
63          return onCompleted(builder.build());
64      }
65  
66      /**
67       * {@inheritDoc}
68       */
69      public void onThrowable(Throwable t) {
70          log.debug(t.getMessage(), t);
71      }
72  
73      /**
74       * Invoked once the HTTP response processing is finished.
75       * <p/>
76       * <p/>
77       * Gets always invoked as last callback method.
78       *
79       * @param response The {@link Response}
80       * @return T Value that will be returned by the associated {@link java.util.concurrent.Future}
81       * @throws Exception if something wrong happens
82       */
83      abstract public T onCompleted(Response response) throws Exception;
84  
85      /**
86       * Invoked when the content (a {@link java.io.File}, {@link String} or {@link java.io.FileInputStream} has been fully
87       * written on the I/O socket.
88       *
89       * @return a {@link com.ning.http.client.AsyncHandler.STATE} telling to CONTINUE or ABORT the current processing.
90       */
91      public STATE onHeaderWriteCompleted() {
92          return STATE.CONTINUE;
93      }
94  
95      /**
96       * Invoked when the content (a {@link java.io.File}, {@link String} or {@link java.io.FileInputStream} has been fully
97       * written on the I/O socket.
98       *
99       * @return a {@link com.ning.http.client.AsyncHandler.STATE} telling to CONTINUE or ABORT the current processing.
100      */
101     public STATE onContentWriteCompleted() {
102         return STATE.CONTINUE;
103     }
104 
105     /**
106      * Invoked when the I/O operation associated with the {@link Request} body as been progressed.
107      *
108      * @param amount  The amount of bytes to transfer.
109      * @param current The amount of bytes transferred
110      * @param total   The total number of bytes transferred
111      * @return a {@link com.ning.http.client.AsyncHandler.STATE} telling to CONTINUE or ABORT the current processing.
112      */
113     public STATE onContentWriteProgress(long amount, long current, long total) {
114         return STATE.CONTINUE;
115     }
116 }
