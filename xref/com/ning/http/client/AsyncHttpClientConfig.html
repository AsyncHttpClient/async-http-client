448     /**
449      * @return <code>true</code> if both the application and reaper thread pools
450      *  haven't yet been shutdown.
451      *
452      * @since 1.7.21
453      */
454     public boolean isValid() {
455         boolean atpRunning = true;
456         try {
457             atpRunning = applicationThreadPool.isShutdown();
458         } catch (Exception ignore) {
459             // isShutdown() will thrown an exception in an EE7 environment
460             // when using a ManagedExecutorService.
461             // When this is the case, we assume it's running.
462         }
463         return atpRunning;
464     }
