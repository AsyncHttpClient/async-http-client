228     public NettyAsyncHttpProvider(AsyncHttpClientConfig config) {
229 
230         if (config.getAsyncHttpProviderConfig() instanceof NettyAsyncHttpProviderConfig) {
231             providerConfig = NettyAsyncHttpProviderConfig.class.cast(config.getAsyncHttpProviderConfig());
232         } else {
233             providerConfig = new NettyAsyncHttpProviderConfig();
234         }
235 
236         if (config.getRequestCompressionLevel() > 0) {
237             LOGGER.warn("Request was enabled but Netty actually doesn't support this feature");
238         }
239 
240         if (providerConfig.getProperty(USE_BLOCKING_IO) != null) {
241             socketChannelFactory = new OioClientSocketChannelFactory(config.executorService());
242             allowReleaseSocketChannelFactory = true;
243         } else {
244             // check if external NioClientSocketChannelFactory is defined
245             Object oo = providerConfig.getProperty(SOCKET_CHANNEL_FACTORY);
246             if (oo instanceof NioClientSocketChannelFactory) {
247                 socketChannelFactory = NioClientSocketChannelFactory.class.cast(oo);
248 
249                 // cannot allow releasing shared channel factory
250                 allowReleaseSocketChannelFactory = false;
251             } else {
252                 ExecutorService e;
253                 Object o = providerConfig.getProperty(BOSS_EXECUTOR_SERVICE);
254                 if (o instanceof ExecutorService) {
255                     e = ExecutorService.class.cast(o);
256                 } else {
257                     e = Executors.newCachedThreadPool();
258                 }
259                 int numWorkers = config.getIoThreadMultiplier() * Runtime.getRuntime().availableProcessors();
260                 log.trace("Number of application's worker threads is {}", numWorkers);
261                 socketChannelFactory = new NioClientSocketChannelFactory(e, config.executorService(), numWorkers);
262                 allowReleaseSocketChannelFactory = true;
263             }
264         }
265 
266         allowStopNettyTimer = providerConfig.getNettyTimer() == null;
267         nettyTimer = allowStopNettyTimer ? newNettyTimer() : providerConfig.getNettyTimer();
268 
269         handshakeTimeoutInMillis = providerConfig.getHandshakeTimeoutInMillis();
270 
271         plainBootstrap = new ClientBootstrap(socketChannelFactory);
272         secureBootstrap = new ClientBootstrap(socketChannelFactory);
273         webSocketBootstrap = new ClientBootstrap(socketChannelFactory);
274         secureWebSocketBootstrap = new ClientBootstrap(socketChannelFactory);
275         this.config = config;
276 
277         configureNetty();
278 
279         // This is dangerous as we can't catch a wrong typed ConnectionsPool
280         ConnectionsPool<String, Channel> cp = (ConnectionsPool<String, Channel>) config.getConnectionsPool();
281         if (cp == null && config.getAllowPoolingConnection()) {
282             cp = new NettyConnectionsPool(this, nettyTimer);
283         } else if (cp == null) {
284             cp = new NonConnectionsPool();
285         }
286         this.connectionsPool = cp;
287 
288         if (config.getMaxTotalConnections() != -1) {
289             trackConnections = true;
290             freeConnections = new Semaphore(config.getMaxTotalConnections());
291         } else {
292             trackConnections = false;
293         }
294 
295         useRawUrl = config.isUseRawUrl();
296         disableZeroCopy = providerConfig.isDisableZeroCopy();
297     }

314     void configureNetty() {
315         if (providerConfig != null) {
316             for (Entry<String, Object> entry : providerConfig.propertiesSet()) {
317                 plainBootstrap.setOption(entry.getKey(), entry.getValue());
318             }
319             configureHttpClientCodec();
320             configureHttpsClientCodec();
321         }
322 
323         plainBootstrap.setPipelineFactory(new ChannelPipelineFactory() {
324 
325             public ChannelPipeline getPipeline() throws Exception {
326                 ChannelPipeline pipeline = pipeline();
327 
328                 pipeline.addLast(HTTP_HANDLER, createHttpClientCodec());
329 
330                 if (config.isCompressionEnabled()) {
331                     pipeline.addLast("inflater", new HttpContentDecompressor());
332                 }
333                 pipeline.addLast("chunkedWriter", new ChunkedWriteHandler());
334                 pipeline.addLast(HTTP_PROCESSOR, NettyAsyncHttpProvider.this);
335                 return pipeline;
336             }
337         });
338         DefaultChannelFuture.setUseDeadLockChecker(false);
339 
340         if (providerConfig != null) {
341             Object value = providerConfig.getProperty(EXECUTE_ASYNC_CONNECT);
342             if (value instanceof Boolean) {
343                 executeConnectAsync = Boolean.class.cast(value);
344             } else if (providerConfig.getProperty(DISABLE_NESTED_REQUEST) != null) {
345                 DefaultChannelFuture.setUseDeadLockChecker(true);
346             }
347         }

350 
351             /* @Override */
352             public ChannelPipeline getPipeline() throws Exception {
353                 ChannelPipeline pipeline = pipeline();
354                 pipeline.addLast(HTTP_HANDLER, createHttpClientCodec());
355                 pipeline.addLast(WS_PROCESSOR, NettyAsyncHttpProvider.this);

357             }
358         });

373     void constructSSLPipeline(final NettyConnectListener<?> cl) {
374 
375         secureBootstrap.setPipelineFactory(new ChannelPipelineFactory() {
376 
377             /* @Override */
378             public ChannelPipeline getPipeline() throws Exception {
379                 ChannelPipeline pipeline = pipeline();
380 
381                 try {
382                     SSLEngine sslEngine = createSSLEngine();
383                     SslHandler sslHandler = handshakeTimeoutInMillis > 0 ? new SslHandler(sslEngine, getDefaultBufferPool(), false, ImmediateExecutor.INSTANCE, nettyTimer,
384                             handshakeTimeoutInMillis) : new SslHandler(sslEngine);
385                     pipeline.addLast(SSL_HANDLER, sslHandler);
386                 } catch (Throwable ex) {
387                     abort(cl.future(), ex);
388                 }
389 
390                 pipeline.addLast(HTTP_HANDLER, createHttpsClientCodec());
391 
392                 if (config.isCompressionEnabled()) {
393                     pipeline.addLast("inflater", new HttpContentDecompressor());
394                 }
395                 pipeline.addLast("chunkedWriter", new ChunkedWriteHandler());
396                 pipeline.addLast(HTTP_PROCESSOR, NettyAsyncHttpProvider.this);
397                 return pipeline;
398             }
399         });
400 
401         secureWebSocketBootstrap.setPipelineFactory(new ChannelPipelineFactory() {
402 
403             /* @Override */
404             public ChannelPipeline getPipeline() throws Exception {
405                 ChannelPipeline pipeline = pipeline();
406 
407                 try {
408                     pipeline.addLast(SSL_HANDLER, new SslHandler(createSSLEngine()));
409                 } catch (Throwable ex) {
410                     abort(cl.future(), ex);
411                 }
412 
413                 pipeline.addLast(HTTP_HANDLER, createHttpsClientCodec());
414                 pipeline.addLast(WS_PROCESSOR, NettyAsyncHttpProvider.this);
415 
416                 return pipeline;
417             }
418         });
419 
420         if (providerConfig != null) {
421             for (Entry<String, Object> entry : providerConfig.propertiesSet()) {
422                 secureBootstrap.setOption(entry.getKey(), entry.getValue());
423                 secureWebSocketBootstrap.setOption(entry.getKey(), entry.getValue());

428     private Channel lookupInCache(URI uri, ProxyServer proxy, ConnectionPoolKeyStrategy strategy) {
429         final Channel channel = connectionsPool.poll(getPoolKey(uri, proxy, strategy));
430 
431         if (channel != null) {
432             log.debug("Using cached Channel {}\n for uri {}\n", channel, uri);

434             try {
435                 // Always make sure the channel who got cached support the proper protocol. It could
436                 // only occurs when a HttpMethod.CONNECT is used against a proxy that requires upgrading from http to

438                 return verifyChannelPipeline(channel, uri.getScheme());
439             } catch (Exception ex) {
440                 log.debug(ex.getMessage(), ex);
441             }
442         }
443         return null;
444     }

462     private Channel verifyChannelPipeline(Channel channel, String scheme) throws IOException, GeneralSecurityException {
463 
464         if (channel.getPipeline().get(SSL_HANDLER) != null && HTTP.equalsIgnoreCase(scheme)) {
465             channel.getPipeline().remove(SSL_HANDLER);
466         } else if (channel.getPipeline().get(HTTP_HANDLER) != null && HTTP.equalsIgnoreCase(scheme)) {
467             return channel;
468         } else if (channel.getPipeline().get(SSL_HANDLER) == null && isSecure(scheme)) {
469             channel.getPipeline().addFirst(SSL_HANDLER, new SslHandler(createSSLEngine()));
470         }
471         return channel;
472     }

474     protected final <T> void writeRequest(final Channel channel, final AsyncHttpClientConfig config, final NettyResponseFuture<T> future) {
475 
476         HttpRequest nettyRequest = future.getNettyRequest();
477         boolean ssl = channel.getPipeline().get(SslHandler.class) != null;
478 
479         try {
480             /**
481              * If the channel is dead because it was pooled and the remote server decided to close it, we just let it go and the channelClosed do it's work.
482              */
483             if (!channel.isOpen() || !channel.isConnected()) {
484                 return;
485             }
486 
487             Body body = null;
488             if (!nettyRequest.getMethod().equals(HttpMethod.CONNECT)) {
489                 BodyGenerator bg = future.getRequest().getBodyGenerator();
490                 if (bg != null) {
491                     // Netty issue with chunking.
492                     if (bg instanceof InputStreamBodyGenerator) {
493                         InputStreamBodyGenerator.class.cast(bg).patchNettyChunkingIssue(true);
494                     }
495 
496                     try {
497                         body = bg.createBody();
498                     } catch (IOException ex) {
499                         throw new IllegalStateException(ex);
500                     }
501                     long length = body.getContentLength();
502                     if (length >= 0) {
503                         nettyRequest.setHeader(HttpHeaders.Names.CONTENT_LENGTH, length);
504                     } else {
505                         nettyRequest.setHeader(HttpHeaders.Names.TRANSFER_ENCODING, HttpHeaders.Values.CHUNKED);
506                     }
507 
508                 } else if (future.getRequest().getParts() != null) {
509                     String contentType = nettyRequest.getHeader(HttpHeaders.Names.CONTENT_TYPE);
510                     String contentLength = nettyRequest.getHeader(HttpHeaders.Names.CONTENT_LENGTH);
511 
512                     long length = -1;
513                     if (contentLength != null) {
514                         length = Long.parseLong(contentLength);
515                     } else {
516                         nettyRequest.addHeader(HttpHeaders.Names.TRANSFER_ENCODING, HttpHeaders.Values.CHUNKED);
517                     }
518 
519                     body = new MultipartBody(future.getRequest().getParts(), contentType, length);
520                 }
521             }
522 
523             if (future.getAsyncHandler() instanceof TransferCompletionHandler) {
524 
525                 FluentCaseInsensitiveStringsMap h = new FluentCaseInsensitiveStringsMap();
526                 for (String s : nettyRequest.getHeaderNames()) {
527                     for (String header : nettyRequest.getHeaders(s)) {
528                         h.add(s, header);
529                     }
530                 }
531 
532                 TransferCompletionHandler.class.cast(future.getAsyncHandler()).transferAdapter(
533                         new NettyTransferAdapter(h, nettyRequest.getContent(), future.getRequest().getFile()));
534             }
535 
536             // Leave it to true.
537             if (future.getAndSetWriteHeaders(true)) {
538                 try {
539                     if (future.getAsyncHandler() instanceof AsyncHandlerExtensions)
540                         AsyncHandlerExtensions.class.cast(future.getAsyncHandler()).onRequestSent();
541 
542                     channel.write(nettyRequest).addListener(new ProgressListener(true, future.getAsyncHandler(), future));
543                 } catch (Throwable cause) {
544                     log.debug(cause.getMessage(), cause);
545                     try {
546                         channel.close();
547                     } catch (RuntimeException ex) {
548                         log.debug(ex.getMessage(), ex);
549                     }
550                     return;
551                 }
552             }
553 
554             if (future.getAndSetWriteBody(true)) {
555                 if (!nettyRequest.getMethod().equals(HttpMethod.CONNECT)) {
556 
557                     if (future.getRequest().getFile() != null) {
558                         final File file = future.getRequest().getFile();
559                         final RandomAccessFile raf = new RandomAccessFile(file, "r");
560 
561                         try {
562                             ChannelFuture writeFuture;
563                             if (disableZeroCopy || ssl) {
564                                 writeFuture = channel.write(new ChunkedFile(raf, 0, raf.length(), MAX_BUFFERED_BYTES));
565                             } else {
566                                 final FileRegion region = new OptimizedFileRegion(raf, 0, raf.length());
567                                 writeFuture = channel.write(region);
568                             }
569                             writeFuture.addListener(new ProgressListener(false, future.getAsyncHandler(), future) {
570                                 public void operationComplete(ChannelFuture cf) {
571                                     try {
572                                         raf.close();
573                                     } catch (IOException e) {
574                                         log.warn("Failed to close request body: {}", e.getMessage(), e);
575                                     }
576                                     super.operationComplete(cf);
577                                 }
578                             });
579                         } catch (IOException ex) {
580                             if (raf != null) {
581                                 try {
582                                     raf.close();
583                                 } catch (IOException e) {
584                                 }
585                             }
586                             throw ex;
587                         }
588                     } else if (body != null) {
589                         final Body b = body;
590 
591                         ChannelFuture writeFuture;
592                         if (disableZeroCopy || ssl || !(body instanceof RandomAccessBody)) {
593                             BodyChunkedInput bodyChunkedInput = new BodyChunkedInput(body);
594                             writeFuture = channel.write(bodyChunkedInput);
595                         } else {
596                             BodyFileRegion bodyFileRegion = new BodyFileRegion((RandomAccessBody) body);
597                             writeFuture = channel.write(bodyFileRegion);
598                         }
599                         writeFuture.addListener(new ProgressListener(false, future.getAsyncHandler(), future) {
600                             public void operationComplete(ChannelFuture cf) {
601                                 try {
602                                     b.close();
603                                 } catch (IOException e) {
604                                     log.warn("Failed to close request body: {}", e.getMessage(), e);
605                                 }
606                                 super.operationComplete(cf);
607                             }
608                         });
609                     }
610                 }
611             }
612         } catch (Throwable ioe) {
613             try {
614                 channel.close();
615             } catch (RuntimeException ex) {
616                 log.debug(ex.getMessage(), ex);
617             }
618         }
619 
620         try {
621             future.touch();
622             int requestTimeoutInMs = AsyncHttpProviderUtils.requestTimeout(config, future.getRequest());
623             TimeoutsHolder timeoutsHolder = new TimeoutsHolder();
624             if (requestTimeoutInMs != -1) {
625                 Timeout requestTimeout = newTimeoutInMs(new RequestTimeoutTimerTask(future, this, timeoutsHolder), requestTimeoutInMs);
626                 timeoutsHolder.requestTimeout = requestTimeout;
627             }
628 
629             int idleConnectionTimeoutInMs = config.getIdleConnectionTimeoutInMs();
630             if (idleConnectionTimeoutInMs != -1 && idleConnectionTimeoutInMs <= requestTimeoutInMs) {
631                 // no need for a idleConnectionTimeout that's less than the requestTimeoutInMs
632                 Timeout idleConnectionTimeout = newTimeoutInMs(new IdleConnectionTimeoutTimerTask(future, this, timeoutsHolder, requestTimeoutInMs, idleConnectionTimeoutInMs),
633                         idleConnectionTimeoutInMs);
634                 timeoutsHolder.idleConnectionTimeout = idleConnectionTimeout;
635             }
636             future.setTimeoutsHolder(timeoutsHolder);
637 
638         } catch (RejectedExecutionException ex) {
639             abort(future, ex);
640         }
641     }

659     private static HttpRequest construct(AsyncHttpClientConfig config, Request request, HttpMethod m, URI uri, ChannelBuffer buffer, ProxyServer proxyServer) throws IOException {
660 
661         String host = null;
662 
663         if (request.getVirtualHost() != null) {
664             host = request.getVirtualHost();
665         } else {
666             host = AsyncHttpProviderUtils.getHost(uri);
667         }
668 
669         HttpRequest nettyRequest;
670         if (m.equals(HttpMethod.CONNECT)) {
671             nettyRequest = new DefaultHttpRequest(HttpVersion.HTTP_1_0, m, AsyncHttpProviderUtils.getAuthority(uri));
672         } else {
673             String path = null;
674             if (proxyServer != null && !(isSecure(uri) && config.isUseRelativeURIsWithSSLProxies()))
675                 path = uri.toString();
676             else if (uri.getRawQuery() != null)
677                 path = uri.getRawPath() + "?" + uri.getRawQuery();
678             else
679                 path = uri.getRawPath();
680             nettyRequest = new DefaultHttpRequest(HttpVersion.HTTP_1_1, m, path);
681         }
682         boolean webSocket = isWebSocket(uri.getScheme());
683         if (!m.equals(HttpMethod.CONNECT) && webSocket) {
684             nettyRequest.addHeader(HttpHeaders.Names.UPGRADE, HttpHeaders.Values.WEBSOCKET);
685             nettyRequest.addHeader(HttpHeaders.Names.CONNECTION, HttpHeaders.Values.UPGRADE);
686             nettyRequest.addHeader(HttpHeaders.Names.ORIGIN, "http://" + uri.getHost() + ":" + uri.getPort());
687             nettyRequest.addHeader(HttpHeaders.Names.SEC_WEBSOCKET_KEY, WebSocketUtil.getKey());
688             nettyRequest.addHeader(HttpHeaders.Names.SEC_WEBSOCKET_VERSION, "13");
689         }
690 
691         if (host != null) {
692             if (request.getVirtualHost() != null || uri.getPort() == -1) {
693                 nettyRequest.setHeader(HttpHeaders.Names.HOST, host);
694             } else {
695                 nettyRequest.setHeader(HttpHeaders.Names.HOST, host + ":" + uri.getPort());
696             }
697         } else {
698             host = "127.0.0.1";
699         }
700 
701         if (!m.equals(HttpMethod.CONNECT)) {
702             for (Entry<String, List<String>> header : request.getHeaders()) {
703                 String name = header.getKey();
704                 if (!HttpHeaders.Names.HOST.equalsIgnoreCase(name)) {
705                     for (String value : header.getValue()) {
706                         nettyRequest.addHeader(name, value);
707                     }
708                 }
709             }
710 
711             if (config.isCompressionEnabled()) {
712                 nettyRequest.setHeader(HttpHeaders.Names.ACCEPT_ENCODING, GZIP_DEFLATE);
713             }
714         } else {
715             List<String> auth = request.getHeaders().get(HttpHeaders.Names.PROXY_AUTHORIZATION);
716             if (isNTLM(auth)) {
717                 nettyRequest.addHeader(HttpHeaders.Names.PROXY_AUTHORIZATION, auth.get(0));
718             }
719         }
720         Realm realm = request.getRealm() != null ? request.getRealm() : config.getRealm();
721 
722         if (realm != null && realm.getUsePreemptiveAuth()) {
723 
724             String domain = realm.getNtlmDomain();
725             if (proxyServer != null && proxyServer.getNtlmDomain() != null) {
726                 domain = proxyServer.getNtlmDomain();
727             }
728 
729             String authHost = realm.getNtlmHost();
730             if (proxyServer != null && proxyServer.getHost() != null) {
731                 host = proxyServer.getHost();
732             }
733 
734             switch (realm.getAuthScheme()) {
735             case BASIC:
736                 nettyRequest.addHeader(HttpHeaders.Names.AUTHORIZATION, AuthenticatorUtils.computeBasicAuthentication(realm));
737                 break;
738             case DIGEST:
739                 if (isNonEmpty(realm.getNonce())) {
740                     try {
741                         nettyRequest.addHeader(HttpHeaders.Names.AUTHORIZATION, AuthenticatorUtils.computeDigestAuthentication(realm));
742                     } catch (NoSuchAlgorithmException e) {
743                         throw new SecurityException(e);
744                     }
745                 }
746                 break;
747             case NTLM:
748                 try {
749                     nettyRequest.addHeader(HttpHeaders.Names.AUTHORIZATION, ntlmEngine.generateType1Msg("NTLM " + domain, authHost));
750                 } catch (NTLMEngineException e) {
751                     IOException ie = new IOException();
752                     ie.initCause(e);
753                     throw ie;
754                 }
755                 break;
756             case KERBEROS:
757             case SPNEGO:
758                 String challengeHeader = null;
759                 String server = proxyServer == null ? host : proxyServer.getHost();
760                 try {
761                     challengeHeader = getSpnegoEngine().generateToken(server);
762                 } catch (Throwable e) {
763                     IOException ie = new IOException();
764                     ie.initCause(e);
765                     throw ie;
766                 }
767                 nettyRequest.addHeader(HttpHeaders.Names.AUTHORIZATION, "Negotiate " + challengeHeader);
768                 break;
769             case NONE:
770                 break;
771             default:
772                 throw new IllegalStateException(String.format("Invalid Authentication %s", realm.toString()));
773             }
774         }
775 
776         if (!webSocket && !request.getHeaders().containsKey(HttpHeaders.Names.CONNECTION)) {
777             nettyRequest.setHeader(HttpHeaders.Names.CONNECTION, AsyncHttpProviderUtils.keepAliveHeaderValue(config));
778         }
779 
780         if (proxyServer != null) {
781             if (!request.getHeaders().containsKey("Proxy-Connection")) {
782                 nettyRequest.setHeader("Proxy-Connection", AsyncHttpProviderUtils.keepAliveHeaderValue(config));
783             }
784 
785             if (proxyServer.getPrincipal() != null) {
786                 if (isNonEmpty(proxyServer.getNtlmDomain())) {
787 
788                     List<String> auth = request.getHeaders().get(HttpHeaders.Names.PROXY_AUTHORIZATION);
789                     if (!isNTLM(auth)) {
790                         try {
791                             String msg = ntlmEngine.generateType1Msg(proxyServer.getNtlmDomain(), proxyServer.getHost());
792                             nettyRequest.setHeader(HttpHeaders.Names.PROXY_AUTHORIZATION, "NTLM " + msg);
793                         } catch (NTLMEngineException e) {
794                             IOException ie = new IOException();
795                             ie.initCause(e);
796                             throw ie;
797                         }
798                     }
799                 } else {
800                     nettyRequest.setHeader(HttpHeaders.Names.PROXY_AUTHORIZATION, AuthenticatorUtils.computeBasicAuthentication(proxyServer));
801                 }
802             }
803         }
804 
805         // Add default accept headers.
806         if (!request.getHeaders().containsKey(HttpHeaders.Names.ACCEPT)) {
807             nettyRequest.setHeader(HttpHeaders.Names.ACCEPT, "*/*");
808         }
809 
810         String userAgentHeader = request.getHeaders().getFirstValue(HttpHeaders.Names.USER_AGENT);
811         if (userAgentHeader != null) {
812             nettyRequest.setHeader(HttpHeaders.Names.USER_AGENT, userAgentHeader);
813         } else if (config.getUserAgent() != null) {
814             nettyRequest.setHeader(HttpHeaders.Names.USER_AGENT, config.getUserAgent());
815         } else {
816             nettyRequest.setHeader(HttpHeaders.Names.USER_AGENT, AsyncHttpProviderUtils.constructUserAgent(NettyAsyncHttpProvider.class));
817         }
818 
819         if (!m.equals(HttpMethod.CONNECT)) {
820             if (isNonEmpty(request.getCookies())) {
821                 nettyRequest.setHeader(HttpHeaders.Names.COOKIE, CookieEncoder.encode(request.getCookies()));
822             }
823 
824             String bodyCharset = request.getBodyEncoding() == null ? DEFAULT_CHARSET : request.getBodyEncoding();
825 
826             // We already have processed the body.
827             if (buffer != null && buffer.writerIndex() != 0) {
828                 nettyRequest.setHeader(HttpHeaders.Names.CONTENT_LENGTH, buffer.writerIndex());
829                 nettyRequest.setContent(buffer);
830             } else if (request.getByteData() != null) {
831                 nettyRequest.setHeader(HttpHeaders.Names.CONTENT_LENGTH, String.valueOf(request.getByteData().length));
832                 nettyRequest.setContent(ChannelBuffers.wrappedBuffer(request.getByteData()));
833             } else if (request.getStringData() != null) {
834                 byte[] bytes = request.getStringData().getBytes(bodyCharset);
835                 nettyRequest.setHeader(HttpHeaders.Names.CONTENT_LENGTH, String.valueOf(bytes.length));
836                 nettyRequest.setContent(ChannelBuffers.wrappedBuffer(bytes));
837             } else if (request.getStreamData() != null) {
838                 int[] lengthWrapper = new int[1];
839                 byte[] bytes = AsyncHttpProviderUtils.readFully(request.getStreamData(), lengthWrapper);
840                 int length = lengthWrapper[0];
841                 nettyRequest.setHeader(HttpHeaders.Names.CONTENT_LENGTH, String.valueOf(length));
842                 nettyRequest.setContent(ChannelBuffers.wrappedBuffer(bytes, 0, length));
843             } else if (isNonEmpty(request.getParams())) {
844                 StringBuilder sb = new StringBuilder();
845                 for (final Entry<String, List<String>> paramEntry : request.getParams()) {
846                     final String key = paramEntry.getKey();
847                     for (final String value : paramEntry.getValue()) {
848                         if (sb.length() > 0) {
849                             sb.append("&");
850                         }
851                         UTF8UrlEncoder.appendEncoded(sb, key);
852                         sb.append("=");
853                         UTF8UrlEncoder.appendEncoded(sb, value);
854                     }
855                 }
856                 nettyRequest.setHeader(HttpHeaders.Names.CONTENT_LENGTH, String.valueOf(sb.length()));
857                 nettyRequest.setContent(ChannelBuffers.wrappedBuffer(sb.toString().getBytes(bodyCharset)));
858 
859                 if (!request.getHeaders().containsKey(HttpHeaders.Names.CONTENT_TYPE)) {
860                     nettyRequest.setHeader(HttpHeaders.Names.CONTENT_TYPE, HttpHeaders.Values.APPLICATION_X_WWW_FORM_URLENCODED);
861                 }
862 
863             } else if (request.getParts() != null) {
864                 MultipartRequestEntity mre = AsyncHttpProviderUtils.createMultipartRequestEntity(request.getParts(), request.getHeaders());
865 
866                 nettyRequest.setHeader(HttpHeaders.Names.CONTENT_TYPE, mre.getContentType());
867                 long contentLength = mre.getContentLength();
868                 if (contentLength >= 0) {
869                     nettyRequest.setHeader(HttpHeaders.Names.CONTENT_LENGTH, String.valueOf(contentLength));
870                 }
871 
872             } else if (request.getEntityWriter() != null) {
873                 int length = (int) request.getContentLength();
874 
875                 if (length == -1) {
876                     length = MAX_BUFFERED_BYTES;
877                 }
878 
879                 ChannelBuffer b = ChannelBuffers.dynamicBuffer(length);
880                 request.getEntityWriter().writeEntity(new ChannelBufferOutputStream(b));
881                 nettyRequest.setHeader(HttpHeaders.Names.CONTENT_LENGTH, b.writerIndex());
882                 nettyRequest.setContent(b);
883             } else if (request.getFile() != null) {
884                 File file = request.getFile();
885                 if (!file.isFile()) {
886                     throw new IOException(String.format("File %s is not a file or doesn't exist", file.getAbsolutePath()));
887                 }
888                 nettyRequest.setHeader(HttpHeaders.Names.CONTENT_LENGTH, file.length());
889             }
890         }
891         return nettyRequest;
892     }

942     private <T> NettyResponseFuture<T> buildNettyResponseFutureWithCachedChannel(Request request, AsyncHandler<T> asyncHandler, NettyResponseFuture<T> f, ProxyServer proxyServer,
943             URI uri, ChannelBuffer bufferedBytes, int maxTry) throws IOException {
944 
945         for (int i = 0; i < maxTry; i++) {
946             if (maxTry == 0)
947                 return null;
948 
949             Channel channel = null;
950             if (f != null && f.reuseChannel() && f.channel() != null) {
951                 channel = f.channel();
952             } else {
953                 channel = lookupInCache(uri, proxyServer, request.getConnectionPoolKeyStrategy());
954             }
955 
956             if (channel == null)
957                 return null;
958             else {
959                 HttpRequest nettyRequest = null;
960 
961                 if (f == null) {
962                     nettyRequest = buildRequest(config, request, uri, false, bufferedBytes, proxyServer);
963                     f = newFuture(uri, request, asyncHandler, nettyRequest, config, this, proxyServer);
964                 } else if (i == 0) {
965                     // only build request on first try
966                     nettyRequest = buildRequest(config, request, uri, f.isConnectAllowed(), bufferedBytes, proxyServer);
967                     f.setNettyRequest(nettyRequest);
968                 }
969                 f.setState(NettyResponseFuture.STATE.POOLED);
970                 f.attachChannel(channel, false);
971 
972                 if (channel.isOpen() && channel.isConnected()) {
973                     f.channel().getPipeline().getContext(NettyAsyncHttpProvider.class).setAttachment(f);
974                     return f;
975                 } else
976                     // else, channel was closed by the server since we fetched it from the pool, starting over
977                     f.attachChannel(null);
978             }
979         }
980         return null;
981     }

983     private <T> ListenableFuture<T> doConnect(final Request request, final AsyncHandler<T> asyncHandler, NettyResponseFuture<T> f, boolean useCache, boolean asyncConnect,
984             boolean reclaimCache) throws IOException {
985 
986         if (isClose()) {
987             throw new IOException("Closed");
988         }
989 
990         if (request.getUrl().startsWith(WEBSOCKET) && !validateWebSocketRequest(request, asyncHandler)) {
991             throw new IOException("WebSocket method must be a GET");
992         }
993 
994         ProxyServer proxyServer = ProxyUtils.getProxyServer(config, request);
995 
996         boolean resultOfAConnect = f != null && f.getNettyRequest() != null && f.getNettyRequest().getMethod().equals(HttpMethod.CONNECT);
997         boolean useProxy = proxyServer != null && !resultOfAConnect;
998 
999         URI uri;
1000         if (useRawUrl) {
1001             uri = request.getRawURI();
1002         } else {
1003             uri = request.getURI();
1004         }
1005         ChannelBuffer bufferedBytes = null;
1006         if (f != null && f.getRequest().getFile() == null && !f.getNettyRequest().getMethod().getName().equals(HttpMethod.CONNECT.getName())) {
1007             bufferedBytes = f.getNettyRequest().getContent();
1008         }
1009 
1010         boolean useSSl = isSecure(uri) && !useProxy;
1011 
1012         if (useCache) {
1013             // 3 tentatives
1014             NettyResponseFuture<T> connectedFuture = buildNettyResponseFutureWithCachedChannel(request, asyncHandler, f, proxyServer, uri, bufferedBytes, 3);
1015 
1016             if (connectedFuture != null) {
1017                 log.debug("\nUsing cached Channel {}\n for request \n{}\n", connectedFuture.channel(), connectedFuture.getNettyRequest());
1018 
1019                 try {
1020                     writeRequest(connectedFuture.channel(), config, connectedFuture);
1021                 } catch (Exception ex) {
1022                     log.debug("writeRequest failure", ex);
1023                     if (useSSl && ex.getMessage() != null && ex.getMessage().contains("SSLEngine")) {
1024                         log.debug("SSLEngine failure", ex);
1025                         connectedFuture = null;
1026                     } else {
1027                         try {
1028                             asyncHandler.onThrowable(ex);
1029                         } catch (Throwable t) {
1030                             log.warn("doConnect.writeRequest()", t);
1031                         }
1032                         IOException ioe = new IOException(ex.getMessage());
1033                         ioe.initCause(ex);
1034                         throw ioe;
1035                     }
1036                 }
1037                 return connectedFuture;
1038             }
1039         }
1040 
1041         // Do not throw an exception when we need an extra connection for a redirect.
1042         if (!reclaimCache && !connectionsPool.canCacheConnection()) {
1043             IOException ex = new IOException(String.format("Too many connections %s", config.getMaxTotalConnections()));
1044             try {
1045                 asyncHandler.onThrowable(ex);
1046             } catch (Throwable t) {
1047                 log.warn("!connectionsPool.canCacheConnection()", t);
1048             }
1049             throw ex;
1050         }
1051 
1052         boolean acquiredConnection = false;
1053 
1054         if (trackConnections) {
1055             if (!reclaimCache) {
1056                 if (!freeConnections.tryAcquire()) {
1057                     IOException ex = new IOException(String.format("Too many connections %s", config.getMaxTotalConnections()));
1058                     try {
1059                         asyncHandler.onThrowable(ex);
1060                     } catch (Throwable t) {
1061                         log.warn("!connectionsPool.canCacheConnection()", t);
1062                     }
1063                     throw ex;
1064                 } else {
1065                     acquiredConnection = true;
1066                 }
1067             }
1068         }
1069 
1070         NettyConnectListener<T> c = new NettyConnectListener.Builder<T>(config, request, asyncHandler, f, this, bufferedBytes).build(uri);
1071 
1072         if (useSSl) {
1073             constructSSLPipeline(c);
1074         }
1075 
1076         ChannelFuture channelFuture;
1077         ClientBootstrap bootstrap = (request.getUrl().startsWith(WEBSOCKET) && !useProxy) ? (useSSl ? secureWebSocketBootstrap : webSocketBootstrap) : (useSSl ? secureBootstrap
1078                 : plainBootstrap);
1079         bootstrap.setOption("connectTimeoutMillis", config.getConnectionTimeoutInMs());
1080 
1081         // Do no enable this with win.
1082         if (!System.getProperty("os.name").toLowerCase(Locale.ENGLISH).contains("win")) {
1083             bootstrap.setOption("reuseAddress", providerConfig.getProperty(REUSE_ADDRESS));
1084         }
1085 
1086         try {
1087             InetSocketAddress remoteAddress;
1088             if (request.getInetAddress() != null) {
1089                 remoteAddress = new InetSocketAddress(request.getInetAddress(), AsyncHttpProviderUtils.getPort(uri));
1090             } else if (!useProxy) {
1091                 remoteAddress = new InetSocketAddress(AsyncHttpProviderUtils.getHost(uri), AsyncHttpProviderUtils.getPort(uri));
1092             } else {
1093                 remoteAddress = new InetSocketAddress(proxyServer.getHost(), proxyServer.getPort());
1094             }
1095 
1096             if (request.getLocalAddress() != null) {
1097                 channelFuture = bootstrap.connect(remoteAddress, new InetSocketAddress(request.getLocalAddress(), 0));
1098             } else {
1099                 channelFuture = bootstrap.connect(remoteAddress);
1100             }
1101 
1102         } catch (Throwable t) {
1103             if (acquiredConnection) {
1104                 freeConnections.release();
1105             }
1106             abort(c.future(), t.getCause() == null ? t : t.getCause());
1107             return c.future();
1108         }
1109 
1110         boolean directInvokation = !(IN_IO_THREAD.get() && DefaultChannelFuture.isUseDeadLockChecker());
1111 
1112         if (directInvokation && !asyncConnect && request.getFile() == null) {
1113             int timeOut = config.getConnectionTimeoutInMs() > 0 ? config.getConnectionTimeoutInMs() : Integer.MAX_VALUE;
1114             if (!channelFuture.awaitUninterruptibly(timeOut, TimeUnit.MILLISECONDS)) {
1115                 if (acquiredConnection) {
1116                     freeConnections.release();
1117                 }
1118                 channelFuture.cancel();
1119                 abort(c.future(), new ConnectException(String.format("Connect operation to %s timeout %s", uri, timeOut)));
1120             }
1121 
1122             try {
1123                 c.operationComplete(channelFuture);
1124             } catch (Exception e) {
1125                 if (acquiredConnection) {
1126                     freeConnections.release();
1127                 }
1128                 IOException ioe = new IOException(e.getMessage());
1129                 ioe.initCause(e);
1130                 try {
1131                     asyncHandler.onThrowable(ioe);
1132                 } catch (Throwable t) {
1133                     log.warn("c.operationComplete()", t);
1134                 }
1135                 throw ioe;
1136             }
1137         } else {
1138             channelFuture.addListener(c);
1139         }
1140 
1141         log.debug("\nNon cached request \n{}\n\nusing Channel \n{}\n", c.future().getNettyRequest(), channelFuture.getChannel());
1142 
1143         if (!c.future().isCancelled() || !c.future().isDone()) {
1144             openChannels.add(channelFuture.getChannel());
1145             c.future().attachChannel(channelFuture.getChannel(), false);
1146         } else {
1147             if (acquiredConnection) {
1148                 freeConnections.release();
1149             }
1150         }
1151         return c.future();
1152     }

1154     protected static int requestTimeoutInMs(AsyncHttpClientConfig config, PerRequestConfig perRequestConfig) {

1490

1619                             ; // Swallow.
1620                         }
1621                         return;

1749             // The write operation failed. If the channel was cached, it means it got asynchronously closed.
1750             // Let's retry a second time.
1751             Throwable cause = cf.getCause();
1752             if (cause != null && future.getState() != NettyResponseFuture.STATE.NEW) {
1753 
1754                 if (cause instanceof IllegalStateException) {
1755                     log.debug(cause.getMessage(), cause);

1765

1767                         log.debug(cf.getCause() == null ? "" : cf.getCause().getMessage(), cf.getCause());
1768                     }
1769 
1770                     try {
1771                         cf.getChannel().close();
1772                     } catch (RuntimeException ex) {
1773                         log.debug(ex.getMessage(), ex);
1774                     }

1777                     future.abort(cause);
1778                 }
1779                 return;
1780             }

1785              * causing unpredictable behavior.
1786              */
