102     /**
103      * Returns the response for the given message.
104      *
105      * @param message  the message that was received from the server.
106      * @param username the username to authenticate with.
107      * @param password the password to authenticate with.
108      * @param host     The host.
109      * @param domain   the NT domain to authenticate in.
110      * @return The response.
111      * @throws NTLMEngineException If the messages cannot be retrieved.
112      */
113     final String getResponseFor(String message, String username, String password,
114                                 String host, String domain) throws NTLMEngineException {
115 
116         final String response;
117         if (isNonEmpty(message)) {
118             Type2Message t2m = new Type2Message(message);
119             response = getType3Message(username, password, host, domain, t2m.getChallenge(), t2m
120                     .getFlags(), t2m.getTarget(), t2m.getTargetInfo());
121         } else {
122             response = getType1Message(host, domain);
123         }
124         return response;
125     }

207     private static int readULong(byte[] src, int index) throws NTLMEngineException {
208         if (src.length < index + 4)
209             throw new NTLMEngineException("NTLM authentication - buffer too small for DWORD");
210         return (src[index] & 0xff) | ((src[index + 1] & 0xff) << 8)
211                 | ((src[index + 2] & 0xff) << 16) | ((src[index + 3] & 0xff) << 24);
212     }
213 
214     private static int readUShort(byte[] src, int index) throws NTLMEngineException {
215         if (src.length < index + 2)
216             throw new NTLMEngineException("NTLM authentication - buffer too small for WORD");
217         return (src[index] & 0xff) | ((src[index + 1] & 0xff) << 8);
218     }
219 
220     private static byte[] readSecurityBuffer(byte[] src, int index) throws NTLMEngineException {
221         int length = readUShort(src, index);
222         int offset = readULong(src, index + 4);
223         if (src.length < offset + length)
224             throw new NTLMEngineException(
225                     "NTLM authentication - buffer too small for data item");
226         byte[] buffer = new byte[length];
227         System.arraycopy(src, offset, buffer, 0, length);
228         return buffer;
229     }

231     /**
232      * Calculate a challenge block
233      */
234     private static byte[] makeRandomChallenge() throws NTLMEngineException {
235         if (RND_GEN == null) {
236             throw new NTLMEngineException("Random generator not available");
237         }
238         byte[] rval = new byte[8];
239         synchronized (RND_GEN) {
240             RND_GEN.nextBytes(rval);
241         }
242         return rval;
243     }

245     /**
246      * Calculate an NTLM2 challenge block
247      */
248     private static byte[] makeNTLM2RandomChallenge() throws NTLMEngineException {
249         if (RND_GEN == null) {
250             throw new NTLMEngineException("Random generator not available");
251         }
252         byte[] rval = new byte[24];
253         synchronized (RND_GEN) {
254             RND_GEN.nextBytes(rval);
255         }
256         // 8-byte challenge, padded with zeros to 24 bytes.
257         Arrays.fill(rval, 8, 24, (byte) 0x00);
258         return rval;
259     }

327     /**
328      * Calculates the NTLM2 Session Response for the given challenge, using the
329      * specified password and client challenge.
330      *
331      * @param password        The user's password.
332      * @param challenge       The Type 2 challenge from the server.
333      * @param clientChallenge The random 8-byte client challenge.
334      * @return The NTLM2 Session Response. This is placed in the NTLM response
335      *         field of the Type 3 message; the LM response field contains the
336      *         client challenge, null-padded to 24 bytes.
337      */
338     static byte[] getNTLM2SessionResponse(String password, byte[] challenge,
339                                           byte[] clientChallenge) throws NTLMEngineException {
340         try {
341             byte[] ntlmHash = ntlmHash(password);
342 
343             // Look up MD5 algorithm (was necessary on jdk 1.4.2)
344             // This used to be needed, but java 1.5.0_07 includes the MD5
345             // algorithm (finally)
346             // Class x = Class.forName("gnu.crypto.hash.MD5");
347             // Method updateMethod = x.getMethod("update",new
348             // Class[]{byte[].class});
349             // Method digestMethod = x.getMethod("digest",new Class[0]);
350             // Object mdInstance = x.newInstance();
351             // updateMethod.invoke(mdInstance,new Object[]{challenge});
352             // updateMethod.invoke(mdInstance,new Object[]{clientChallenge});
353             // byte[] digest = (byte[])digestMethod.invoke(mdInstance,new
354             // Object[0]);
355 
356             MessageDigest md5 = MessageDigest.getInstance("MD5");
357             md5.update(challenge);
358             md5.update(clientChallenge);
359             byte[] digest = md5.digest();
360 
361             byte[] sessionHash = new byte[8];
362             System.arraycopy(digest, 0, sessionHash, 0, 8);
363             return lmResponse(ntlmHash, sessionHash);
364         } catch (Exception e) {
365             if (e instanceof NTLMEngineException)
366                 throw (NTLMEngineException) e;
367             throw new NTLMEngineException(e.getMessage(), e);
368         }
369     }

371     /**
372      * Creates the LM Hash of the user's password.
373      *
374      * @param password The password.
375      * @return The LM Hash of the given password, used in the calculation of the
376      *         LM Response.
377      */
378     private static byte[] lmHash(String password) throws NTLMEngineException {
379         try {
380             byte[] oemPassword = password.toUpperCase(Locale.ENGLISH).getBytes("US-ASCII");
381             int length = Math.min(oemPassword.length, 14);
382             byte[] keyBytes = new byte[14];
383             System.arraycopy(oemPassword, 0, keyBytes, 0, length);
384             Key lowKey = createDESKey(keyBytes, 0);
385             Key highKey = createDESKey(keyBytes, 7);
386             byte[] magicConstant = "KGS!@#$%".getBytes("US-ASCII");
387             Cipher des = Cipher.getInstance("DES/ECB/NoPadding");
388             des.init(Cipher.ENCRYPT_MODE, lowKey);
389             byte[] lowHash = des.doFinal(magicConstant);
390             des.init(Cipher.ENCRYPT_MODE, highKey);
391             byte[] highHash = des.doFinal(magicConstant);
392             byte[] lmHash = new byte[16];
393             System.arraycopy(lowHash, 0, lmHash, 0, 8);
394             System.arraycopy(highHash, 0, lmHash, 8, 8);
395             return lmHash;
396         } catch (Exception e) {
397             throw new NTLMEngineException(e.getMessage(), e);
398         }
399     }

401     /**
402      * Creates the NTLM Hash of the user's password.
403      *
404      * @param password The password.
405      * @return The NTLM Hash of the given password, used in the calculation of
406      *         the NTLM Response and the NTLMv2 and LMv2 Hashes.
407      */
408     private static byte[] ntlmHash(String password) throws NTLMEngineException {
409         try {
410             byte[] unicodePassword = password.getBytes("UnicodeLittleUnmarked");
411             MD4 md4 = new MD4();
412             md4.update(unicodePassword);
413             return md4.getOutput();
414         } catch (java.io.UnsupportedEncodingException e) {
415             throw new NTLMEngineException("Unicode not supported: " + e.getMessage(), e);
416         }
417     }

419     /**
420      * Creates the NTLMv2 Hash of the user's password.
421      *
422      * @param target   The authentication target (i.e., domain).
423      * @param user     The username.
424      * @param password The password.
425      * @return The NTLMv2 Hash, used in the calculation of the NTLMv2 and LMv2
426      *         Responses.
427      */
428     private static byte[] ntlmv2Hash(String target, String user, String password)
429             throws NTLMEngineException {
430         try {
431             byte[] ntlmHash = ntlmHash(password);
432             HMACMD5 hmacMD5 = new HMACMD5(ntlmHash);
433             // Upper case username, mixed case target!!
434             hmacMD5.update(user.toUpperCase(Locale.ENGLISH).getBytes("UnicodeLittleUnmarked"));
435             hmacMD5.update(target.getBytes("UnicodeLittleUnmarked"));
436             return hmacMD5.getOutput();
437         } catch (java.io.UnsupportedEncodingException e) {
438             throw new NTLMEngineException("Unicode not supported! " + e.getMessage(), e);
439         }
440     }

442     /**
443      * Creates the LM Response from the given hash and Type 2 challenge.
444      *
445      * @param hash      The LM or NTLM Hash.
446      * @param challenge The server challenge from the Type 2 message.
447      * @return The response (either LM or NTLM, depending on the provided hash).
448      */
449     private static byte[] lmResponse(byte[] hash, byte[] challenge) throws NTLMEngineException {
450         try {
451             byte[] keyBytes = new byte[21];
452             System.arraycopy(hash, 0, keyBytes, 0, 16);
453             Key lowKey = createDESKey(keyBytes, 0);
454             Key middleKey = createDESKey(keyBytes, 7);
455             Key highKey = createDESKey(keyBytes, 14);
456             Cipher des = Cipher.getInstance("DES/ECB/NoPadding");
457             des.init(Cipher.ENCRYPT_MODE, lowKey);
458             byte[] lowResponse = des.doFinal(challenge);
459             des.init(Cipher.ENCRYPT_MODE, middleKey);
460             byte[] middleResponse = des.doFinal(challenge);
461             des.init(Cipher.ENCRYPT_MODE, highKey);
462             byte[] highResponse = des.doFinal(challenge);
463             byte[] lmResponse = new byte[24];
464             System.arraycopy(lowResponse, 0, lmResponse, 0, 8);
465             System.arraycopy(middleResponse, 0, lmResponse, 8, 8);
466             System.arraycopy(highResponse, 0, lmResponse, 16, 8);
467             return lmResponse;
468         } catch (Exception e) {
469             throw new NTLMEngineException(e.getMessage(), e);
470         }
471     }

473     /**
474      * Creates the LMv2 Response from the given hash, client data, and Type 2
475      * challenge.
476      *
477      * @param hash       The NTLMv2 Hash.
478      * @param clientData The client data (blob or client challenge).
479      * @param challenge  The server challenge from the Type 2 message.
480      * @return The response (either NTLMv2 or LMv2, depending on the client
481      *         data).
482      */
483     private static byte[] lmv2Response(byte[] hash, byte[] challenge, byte[] clientData)
484             throws NTLMEngineException {
485         HMACMD5 hmacMD5 = new HMACMD5(hash);
486         hmacMD5.update(challenge);
487         hmacMD5.update(clientData);
488         byte[] mac = hmacMD5.getOutput();
489         byte[] lmv2Response = new byte[mac.length + clientData.length];
490         System.arraycopy(mac, 0, lmv2Response, 0, mac.length);
491         System.arraycopy(clientData, 0, lmv2Response, mac.length, clientData.length);
492         return lmv2Response;
493     }

495     /**
496      * Creates the NTLMv2 blob from the given target information block and
497      * client challenge.
498      *
499      * @param targetInformation The target information block from the Type 2 message.
500      * @param clientChallenge   The random 8-byte client challenge.
501      * @return The blob, used in the calculation of the NTLMv2 Response.
502      */
503     private static byte[] createBlob(byte[] clientChallenge, byte[] targetInformation) {
504         byte[] blobSignature = new byte[]{(byte) 0x01, (byte) 0x01, (byte) 0x00, (byte) 0x00};
505         byte[] reserved = new byte[]{(byte) 0x00, (byte) 0x00, (byte) 0x00, (byte) 0x00};
506         byte[] unknown1 = new byte[]{(byte) 0x00, (byte) 0x00, (byte) 0x00, (byte) 0x00};
507         long time = System.currentTimeMillis();
508         time += 11644473600000l; // milliseconds from January 1, 1601 -> epoch.
509         time *= 10000; // tenths of a microsecond.
510         // convert to little-endian byte array.
511         byte[] timestamp = new byte[8];
512         for (int i = 0; i < 8; i++) {
513             timestamp[i] = (byte) time;
514             time >>>= 8;
515         }
516         byte[] blob = new byte[blobSignature.length + reserved.length + timestamp.length + 8
517                 + unknown1.length + targetInformation.length];
518         int offset = 0;
519         System.arraycopy(blobSignature, 0, blob, offset, blobSignature.length);
520         offset += blobSignature.length;
521         System.arraycopy(reserved, 0, blob, offset, reserved.length);
522         offset += reserved.length;
523         System.arraycopy(timestamp, 0, blob, offset, timestamp.length);
524         offset += timestamp.length;
525         System.arraycopy(clientChallenge, 0, blob, offset, 8);
526         offset += 8;
527         System.arraycopy(unknown1, 0, blob, offset, unknown1.length);
528         offset += unknown1.length;
529         System.arraycopy(targetInformation, 0, blob, offset, targetInformation.length);
530         return blob;
531     }

533     /**
534      * Creates a DES encryption key from the given key material.
535      *
536      * @param bytes  A byte array containing the DES key material.
537      * @param offset The offset in the given byte array at which the 7-byte key
538      *               material starts.
539      * @return A DES encryption key created from the key material starting at
540      *         the specified offset in the given byte array.
541      */
542     private static Key createDESKey(byte[] bytes, int offset) {
543         byte[] keyBytes = new byte[7];
544         System.arraycopy(bytes, offset, keyBytes, 0, 7);
545         byte[] material = new byte[8];
546         material[0] = keyBytes[0];
547         material[1] = (byte) (keyBytes[0] << 7 | (keyBytes[1] & 0xff) >>> 1);
548         material[2] = (byte) (keyBytes[1] << 6 | (keyBytes[2] & 0xff) >>> 2);
549         material[3] = (byte) (keyBytes[2] << 5 | (keyBytes[3] & 0xff) >>> 3);
550         material[4] = (byte) (keyBytes[3] << 4 | (keyBytes[4] & 0xff) >>> 4);
551         material[5] = (byte) (keyBytes[4] << 3 | (keyBytes[5] & 0xff) >>> 5);
552         material[6] = (byte) (keyBytes[5] << 2 | (keyBytes[6] & 0xff) >>> 6);
553         material[7] = (byte) (keyBytes[6] << 1);
554         oddParity(material);
555         return new SecretKeySpec(material, "DES");
556     }

558     /**
559      * Applies odd parity to the given byte array.
560      *
561      * @param bytes The data whose parity bits are to be adjusted for odd parity.
562      */
563     private static void oddParity(byte[] bytes) {
564         for (int i = 0; i < bytes.length; i++) {
565             byte b = bytes[i];
566             boolean needsParity = (((b >>> 7) ^ (b >>> 6) ^ (b >>> 5) ^ (b >>> 4) ^ (b >>> 3)
567                     ^ (b >>> 2) ^ (b >>> 1)) & 0x01) == 0;
568             if (needsParity) {
569                 bytes[i] |= (byte) 0x01;
570             } else {
571                 bytes[i] &= (byte) 0xfe;
572             }
573         }
574     }

576     /**
577      * NTLM message generation, base class
578      */
579     static class NTLMMessage {
580         /**
581          * The current response
582          */
583         private byte[] messageContents = null;
584 
585         /**
586          * The current output position
587          */
588         private int currentOutputPosition = 0;
589 
590         /**
591          * Constructor to use when message contents are not yet known
592          */
593         NTLMMessage() {
594         }
595 
596         /**
597          * Constructor to use when message contents are known
598          */
599         NTLMMessage(String messageBody, int expectedType) throws NTLMEngineException {
600             messageContents = Base64.decode(messageBody);
601 
602             // Look for NTLM message
603             if (messageContents.length < SIGNATURE.length)
604                 throw new NTLMEngineException("NTLM message decoding error - packet too short");
605             int i = 0;
606             while (i < SIGNATURE.length) {
607                 if (messageContents[i] != SIGNATURE[i])
608                     throw new NTLMEngineException(
609                             "NTLM message expected - instead got unrecognized bytes");
610                 i++;
611             }
612 
613             // Check to be sure there's a type 2 message indicator next
614             int type = readULong(SIGNATURE.length);
615             if (type != expectedType)
616                 throw new NTLMEngineException("NTLM type " + Integer.toString(expectedType)
617                         + " message expected - instead got type " + Integer.toString(type));
618 
619             currentOutputPosition = messageContents.length;
620         }
621 
622         /**

865             if (getMessageLength() >= 40 + 8) {
866                 byte[] bytes = readSecurityBuffer(40);
867                 if (bytes.length != 0) {

869                 }
870             }

921                      int type2Flags, String target, byte[] targetInformation)
922                 throws NTLMEngineException {

924             this.type2Flags = type2Flags;
925

929             domain = convertDomain(domain);

982             int userLen = userBytes.length;

1079      * Cryptography support - MD4. The following class was based loosely on the
1080      * RFC and on code found at http://www.cs.umd.edu/~harry/jotp/src/md.java.
1081      * Code correctness was verified by looking at MD4.java from the jcifs
1082      * library (http://jcifs.samba.org). It was massaged extensively to the
1083      * final form found here by Karl Wright (kwright@metacarta.com).
1084      */
1085     static class MD4 {
1086         protected int A = 0x67452301;
1087         protected int B = 0xefcdab89;
1088         protected int C = 0x98badcfe;
1089         protected int D = 0x10325476;
1090         protected long count = 0L;
1091         protected byte[] dataBuffer = new byte[64];
1092 
1093         MD4() {
1094         }
1095 
1096         void update(byte[] input) {
1097             // We always deal with 512 bits at a time. Correspondingly, there is
1098             // a buffer 64 bytes long that we write data into until it gets
1099             // full.
1100             int curBufferPos = (int) (count & 63L);
1101             int inputIndex = 0;
1102             while (input.length - inputIndex + curBufferPos >= dataBuffer.length) {
1103                 // We have enough data to do the next step. Do a partial copy
1104                 // and a transform, updating inputIndex and curBufferPos
1105                 // accordingly
1106                 int transferAmt = dataBuffer.length - curBufferPos;
1107                 System.arraycopy(input, inputIndex, dataBuffer, curBufferPos, transferAmt);
1108                 count += transferAmt;
1109                 curBufferPos = 0;
1110                 inputIndex += transferAmt;
1111                 processBuffer();
1112             }
1113 
1114             // If there's anything left, copy it into the buffer and leave it.
1115             // We know there's not enough left to process.
1116             if (inputIndex < input.length) {
1117                 int transferAmt = input.length - inputIndex;
1118                 System.arraycopy(input, inputIndex, dataBuffer, curBufferPos, transferAmt);
1119                 count += transferAmt;
1120             }
1121         }
1122 
1123         byte[] getOutput() {
1124             // Feed pad/length data into engine. This must round out the input
1125             // to a multiple of 512 bits.
1126             int bufferIndex = (int) (count & 63L);
1127             int padLen = (bufferIndex < 56) ? (56 - bufferIndex) : (120 - bufferIndex);
1128             byte[] postBytes = new byte[padLen + 8];
1129             // Leading 0x80, specified amount of zero padding, then length in
1130             // bits.
1131             postBytes[0] = (byte) 0x80;
1132             // Fill out the last 8 bytes with the length
1133             for (int i = 0; i < 8; i++) {
1134                 postBytes[padLen + i] = (byte) ((count * 8) >>> (8 * i));
1135             }
1136 
1137             // Update the engine
1138             update(postBytes);
1139 
1140             // Calculate final result
1141             byte[] result = new byte[16];
1142             writeULong(result, A, 0);
1143             writeULong(result, B, 4);
1144             writeULong(result, C, 8);
1145             writeULong(result, D, 12);
1146             return result;
1147         }
1148 
1149         protected void processBuffer() {
1150             // Convert current buffer to 16 ulongs
1151             int[] d = new int[16];
1152 
1153             for (int i = 0; i < 16; i++) {
1154                 d[i] = (dataBuffer[i * 4] & 0xff) + ((dataBuffer[i * 4 + 1] & 0xff) << 8)
1155                         + ((dataBuffer[i * 4 + 2] & 0xff) << 16)
1156                         + ((dataBuffer[i * 4 + 3] & 0xff) << 24);
1157             }
1158 
1159             // Do a round of processing
1160             int AA = A;
1161             int BB = B;
1162             int CC = C;
1163             int DD = D;
1164             round1(d);
1165             round2(d);
1166             round3(d);
1167             A += AA;
1168             B += BB;
1169             C += CC;
1170             D += DD;
1171 
1172         }
