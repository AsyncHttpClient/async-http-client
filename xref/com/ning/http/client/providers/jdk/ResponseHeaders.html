13  package com.ning.http.client.providers.jdk;
14  
15  import com.ning.http.client.AsyncHttpProvider;
16  import com.ning.http.client.FluentCaseInsensitiveStringsMap;
17  import com.ning.http.client.HttpResponseHeaders;
18  
19  import java.net.HttpURLConnection;
20  import java.net.URI;
21  import java.util.List;
22  import java.util.Map;
23  
24  /**
25   * A class that represent the HTTP headers.
26   */
27  public class ResponseHeaders extends HttpResponseHeaders {
28  
29      private final HttpURLConnection urlConnection;
30      private final FluentCaseInsensitiveStringsMap headers;
31  
32      public ResponseHeaders(URI uri, HttpURLConnection urlConnection, AsyncHttpProvider provider) {
33          super(uri, provider, false);
34          this.urlConnection = urlConnection;
35          headers = computerHeaders();
36      }
37  
38      private FluentCaseInsensitiveStringsMap computerHeaders() {
39          FluentCaseInsensitiveStringsMap h = new FluentCaseInsensitiveStringsMap();
40  
41          Map<String, List<String>> uh = urlConnection.getHeaderFields();
42  
43          for (Map.Entry<String, List<String>> e : uh.entrySet()) {
44              if (e.getKey() != null) {
45                  h.add(e.getKey(), e.getValue());
46              }
47          }
48          return h;
49      }
50  
51      /**
52       * Return the HTTP header
53       *
54       * @return an {@link com.ning.http.client.FluentCaseInsensitiveStringsMap}
55       */
56      @Override
57      public FluentCaseInsensitiveStringsMap getHeaders() {
58          return headers;
59      }
60  }
