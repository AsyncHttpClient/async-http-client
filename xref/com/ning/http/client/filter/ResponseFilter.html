13  package com.ning.http.client.filter;
14  
15  /**
16   * A Filter interface that gets invoked before making the processing of the response bytes. {@link ResponseFilter} are invoked
17   * before the actual response's status code get processed. That means authorization, proxy authentication and redirects
18   * processing hasn't occured when {@link ResponseFilter} gets invoked.
19   */
20  public interface ResponseFilter {
21  
22      /**
23       * An {@link com.ning.http.client.AsyncHttpProvider} will invoke {@link ResponseFilter#filter} and will use the
24       * returned {@link FilterContext#replayRequest()} and {@link FilterContext#getAsyncHandler()} to decide if the response
25       * processing can continue. If {@link FilterContext#replayRequest()} return true, a new request will be made
26       * using {@link FilterContext#getRequest()} and the current response processing will be ignored.
27       *
28       * @param ctx a {@link FilterContext}
29       * @return {@link FilterContext}. The {@link FilterContext} instance may not the same as the original one.
30       * @throws FilterException to interrupt the filter processing.
31       */
32      public FilterContext filter(FilterContext ctx) throws FilterException;
33  
34  }
