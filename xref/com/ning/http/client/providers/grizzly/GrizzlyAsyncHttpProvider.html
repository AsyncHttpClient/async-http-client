206      * {@inheritDoc}
207      */
208     @SuppressWarnings({"unchecked"})
209     public <T> ListenableFuture<T> execute(final Request request,
210                                            final AsyncHandler<T> handler) throws IOException {
211 
212         if (clientTransport.isStopped()) {
213             throw new IOException("AsyncHttpClient has been closed.");
214         }
215         final ProxyServer proxy = ProxyUtils.getProxyServer(clientConfig, request);
216         final GrizzlyResponseFuture<T> future = new GrizzlyResponseFuture<T>(this, request, handler, proxy);
217         future.setDelegate(SafeFutureImpl.<T>create());
218         final CompletionHandler<Connection>  connectHandler = new CompletionHandler<Connection>() {
219             @Override
220             public void cancelled() {
221                 future.cancel(true);
222             }
223 
224             @Override
225             public void failed(final Throwable throwable) {
226                 future.abort(throwable);
227             }
228 
229             @Override
230             public void completed(final Connection c) {
231                 try {
232                     execute(c, request, handler, future);
233                 } catch (Exception e) {
234                     if (e instanceof RuntimeException) {
235                         failed(e);
236                     } else if (e instanceof IOException) {
237                         failed(e);
238                     }
239                     if (LOGGER.isWarnEnabled()) {
240                         LOGGER.warn(e.toString(), e);
241                     }
242                 }
243             }
244 
245             @Override
246             public void updated(final Connection c) {
247                 // no-op
248             }
249         };
250 
251         try {
252             connectionManager.doAsyncTrackedConnection(request, future, connectHandler);
253         } catch (Exception e) {
254             if (e instanceof RuntimeException) {
255                 throw (RuntimeException) e;
256             } else if (e instanceof IOException) {
257                 throw (IOException) e;
258             }
259             if (LOGGER.isWarnEnabled()) {
260                 LOGGER.warn(e.toString(), e);
261             }
262         }
263 
264         return future;
265     }

267     /**
268      * {@inheritDoc}

270     public void close() {
271 
272         try {
273             connectionManager.destroy();
274             clientTransport.shutdownNow();
275             final ExecutorService service = clientConfig.executorService();
276             if (service != null) {
277                 service.shutdown();
278             }
279             if (timeoutExecutor != null) {
280                 timeoutExecutor.stop();
281                 timeoutExecutor.getThreadPool().shutdownNow();
282             }
283         } catch (IOException ignored) { }
284

303     @SuppressWarnings({"unchecked"})
304     protected <T> ListenableFuture<T> execute(final Connection c,
305                                               final Request request,
306                                               final AsyncHandler<T> handler,
307                                               final GrizzlyResponseFuture<T> future)
308     throws IOException {
309 
310         try {
311             if (getHttpTransactionContext(c) == null) {
312                 setHttpTransactionContext(c,
313                         new HttpTransactionContext(future, request, handler));
314             }
315             c.write(request, createWriteCompletionHandler(future));
316         } catch (Exception e) {
317             if (e instanceof RuntimeException) {
318                 throw (RuntimeException) e;
319             } else if (e instanceof IOException) {
320                 throw (IOException) e;
321             }
322             if (LOGGER.isWarnEnabled()) {
323                 LOGGER.warn(e.toString(), e);
324             }
325         }
326 
327         return future;
328     }

331     protected void initializeTransport(final AsyncHttpClientConfig clientConfig) {
332 
333         final FilterChainBuilder fcb = FilterChainBuilder.stateless();
334         fcb.add(new AsyncHttpClientTransportFilter());
335 
336         final int timeout = clientConfig.getRequestTimeoutInMs();
337         if (timeout > 0) {
338             int delay = 500;
339             if (timeout < delay) {
340                 delay = timeout - 10;
341             }
342             timeoutExecutor = IdleTimeoutFilter.createDefaultIdleDelayedExecutor(delay, TimeUnit.MILLISECONDS);
343             timeoutExecutor.start();
344             final IdleTimeoutFilter.TimeoutResolver timeoutResolver =
345                     new IdleTimeoutFilter.TimeoutResolver() {
346                         @Override
347                         public long getTimeout(FilterChainContext ctx) {
348                             final HttpTransactionContext context =
349                                     GrizzlyAsyncHttpProvider.this.getHttpTransactionContext(ctx.getConnection());
350                             if (context != null) {
351                                 if (context.isWSRequest) {
352                                     return clientConfig.getWebSocketIdleTimeoutInMs();
353                                 }
354                                 final PerRequestConfig config = context.request.getPerRequestConfig();
355                                 if (config != null) {
356                                     final long timeout = config.getRequestTimeoutInMs();
357                                     if (timeout > 0) {
358                                         return timeout;
359                                     }
360                                 }
361                             }
362                             return timeout;
363                         }
364                     };
365             final IdleTimeoutFilter timeoutFilter = new IdleTimeoutFilter(timeoutExecutor,
366                     timeoutResolver,
367                     new IdleTimeoutFilter.TimeoutHandler() {
368                         public void onTimeout(Connection connection) {
369                             timeout(connection);
370                         }
371                     });
372             fcb.add(timeoutFilter);
373             resolver = timeoutFilter.getResolver();
374         }
375 
376         SSLContext context = clientConfig.getSSLContext();
377         boolean defaultSecState = (context != null);
378         if (context == null) {
379             try {
380                 context = SslUtils.getSSLContext();
381             } catch (Exception e) {
382                 throw new IllegalStateException(e);
383             }
384         }
385         final SSLEngineConfigurator configurator =
386                 new SSLEngineConfigurator(context,
387                         true,
388                         false,
389                         false);
390         final SwitchingSSLFilter filter = new SwitchingSSLFilter(configurator, defaultSecState);
391         fcb.add(filter);
392         final GrizzlyAsyncHttpProviderConfig providerConfig =
393                 clientConfig.getAsyncHttpProviderConfig() instanceof GrizzlyAsyncHttpProviderConfig ?
394                 (GrizzlyAsyncHttpProviderConfig) clientConfig.getAsyncHttpProviderConfig()
395                 : new GrizzlyAsyncHttpProviderConfig();
396         final AsyncHttpClientEventFilter eventFilter = new
397                 AsyncHttpClientEventFilter(this, (Integer) providerConfig.getProperty(MAX_HTTP_PACKET_HEADER_SIZE));
398         final AsyncHttpClientFilter clientFilter =
399                 new AsyncHttpClientFilter(clientConfig);
400         ContentEncoding[] encodings = eventFilter.getContentEncodings();
401         if (encodings.length > 0) {
402             for (ContentEncoding encoding : encodings) {
403                 eventFilter.removeContentEncoding(encoding);
404             }
405         }
406         if (clientConfig.isCompressionEnabled()) {
407             eventFilter.addContentEncoding(
408                     new GZipContentEncoding(512,
409                                             512,
410                                             new ClientEncodingFilter()));
411         }
412         fcb.add(eventFilter);
413         fcb.add(clientFilter);
414         clientTransport.getAsyncQueueIO().getWriter()
415                        .setMaxPendingBytesPerConnection(AsyncQueueWriter.AUTO_SIZE);
416         final TransportCustomizer customizer = (TransportCustomizer)
417                 providerConfig.getProperty(TRANSPORT_CUSTOMIZER);
418         if (customizer != null) {
419             customizer.customize(clientTransport, fcb);
420         } else {
421             doDefaultTransportConfig();
422         }
423         fcb.add(new WebSocketFilter());
424 
425         clientTransport.setProcessor(fcb.build());
426 
427     }

433     void touchConnection(final Connection c, final Request request) {
434 
435         final PerRequestConfig config = request.getPerRequestConfig();
436         if (config != null) {
437             final long timeout = config.getRequestTimeoutInMs();
438             if (timeout > 0) {
439                 final long newTimeout = System.currentTimeMillis() + timeout;
440                 if (resolver != null) {
441                     resolver.setTimeoutMillis(c, newTimeout);
442                 }
443             }
444         } else {
445             final long timeout = clientConfig.getRequestTimeoutInMs();
446             if (timeout > 0) {
447                 if (resolver != null) {
448                     resolver.setTimeoutMillis(c, System.currentTimeMillis() + timeout);
449                 }
450             }
451         }
452

467     private static boolean linuxSendFileSupported() {
468         final String version = System.getProperty("java.version");
469         if (version.startsWith("1.6")) {
470             int idx = version.indexOf('_');
471             if (idx == -1) {
472                 return false;
473             }
474             final int patchRev = Integer.parseInt(version.substring(idx + 1));
475             return (patchRev >= 18);
476         } else {
477             return version.startsWith("1.7") || version.startsWith("1.8");
478         }

540     static int getPort(final URI uri, final int p) {
541         int port = p;
542         if (port == -1) {
543             final String protocol = uri.getScheme().toLowerCase(Locale.ENGLISH);
544             if ("http".equals(protocol) || "ws".equals(protocol)) {
545                 port = 80;
546             } else if ("https".equals(protocol) || "wss".equals(protocol)) {
547                 port = 443;
548             } else {
549                 throw new IllegalArgumentException("Unknown protocol: " + protocol);
550             }
551         }
552         return port;
553     }

556     @SuppressWarnings({"unchecked"})
557     boolean sendRequest(final FilterChainContext ctx,
558                      final Request request,
559                      final HttpRequestPacket requestPacket)
560     throws IOException {
561 
562         boolean isWriteComplete = true;
563         
564         if (requestHasEntityBody(request)) {
565             final HttpTransactionContext context = getHttpTransactionContext(ctx.getConnection());
566             BodyHandler handler = bodyHandlerFactory.getBodyHandler(request);
567             if (requestPacket.getHeaders().contains(Header.Expect)
568                     && requestPacket.getHeaders().getValue(1).equalsIgnoreCase("100-Continue")) {
569                 handler = new ExpectHandler(handler);
570             }
571             context.bodyHandler = handler;
572             if (LOGGER.isDebugEnabled()) {
573                 LOGGER.debug("REQUEST: " + requestPacket.toString());
574             }
575             isWriteComplete = handler.doHandle(ctx, request, requestPacket);
576         } else {
577             if (LOGGER.isDebugEnabled()) {
578                 LOGGER.debug("REQUEST: " + requestPacket.toString());
579             }
580             ctx.write(requestPacket, ctx.getTransportContext().getCompletionHandler());
581         }
582 
583         
584         return isWriteComplete;
585     }

878             builder.method(request.getMethod());
879             builder.protocol(Protocol.HTTP_1_1);
