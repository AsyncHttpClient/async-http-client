41  /**
42   * This class support asynchronous and synchronous HTTP request.
43   * <p/>
44   * To execute synchronous HTTP request, you just need to do
45   * <blockquote><pre>
46   *    AsyncHttpClient c = new AsyncHttpClient();
47   *    Future<Response> f = c.prepareGet("http://www.ning.com/").execute();
48   * </pre></blockquote
49   * <p/>
50   * The code above will block until the response is fully received. To execute asynchronous HTTP request, you
51   * create an {@link AsyncHandler} or its abstract implementation, {@link com.ning.http.client.AsyncCompletionHandler}
52   * <p/>
53   * <blockquote><pre>
54   *       AsyncHttpClient c = new AsyncHttpClient();
55   *       Future<Response> f = c.prepareGet("http://www.ning.com/").execute(new AsyncCompletionHandler<Response>() &#123;
56   * <p/>
57   *          &#64;Override
58   *          public Response onCompleted(Response response) throws IOException &#123;
59   *               // Do something
60   *              return response;
61   *          &#125;
62   * <p/>
63   *          &#64;Override
64   *          public void onThrowable(Throwable t) &#123;
65   *          &#125;
66   *      &#125;);
67   *      Response response = f.get();
68   * <p/>
69   *      // We are just interested to retrieve the status code.
70   *     Future<Integer> f = c.prepareGet("http://www.ning.com/").execute(new AsyncCompletionHandler<Integer>() &#123;
71   * <p/>
72   *          &#64;Override
73   *          public Integer onCompleted(Response response) throws IOException &#123;
74   *               // Do something
75   *              return response.getStatusCode();
76   *          &#125;
77   * <p/>
78   *          &#64;Override
79   *          public void onThrowable(Throwable t) &#123;
80   *          &#125;
81   *      &#125;);
82   *      Integer statusCode = f.get();
83   * </pre></blockquote
84   * The {@link AsyncCompletionHandler#onCompleted(com.ning.http.client.Response)} will be invoked once the http response has been fully read, which include
85   * the http headers and the response body. Note that the entire response will be buffered in memory.
86   * <p/>
87   * You can also have more control about the how the response is asynchronously processed by using a {@link AsyncHandler}
88   * <blockquote><pre>
89   *      AsyncHttpClient c = new AsyncHttpClient();
90   *      Future<String> f = c.prepareGet("http://www.ning.com/").execute(new AsyncHandler<String>() &#123;
91   *          private StringBuilder builder = new StringBuilder();
92   * <p/>
93   *          &#64;Override
94   *          public STATE onStatusReceived(HttpResponseStatus s) throws Exception &#123;
95   *               // return STATE.CONTINUE or STATE.ABORT
96   *               return STATE.CONTINUE
97   *          }
98   * <p/>
99   *          &#64;Override
100  *          public STATE onHeadersReceived(HttpResponseHeaders bodyPart) throws Exception &#123;
101  *               // return STATE.CONTINUE or STATE.ABORT
102  *               return STATE.CONTINUE
103  * <p/>
104  *          }
105  *          &#64;Override
106  * <p/>
107  *          public STATE onBodyPartReceived(HttpResponseBodyPart bodyPart) throws Exception &#123;
108  *               builder.append(new String(bodyPart));
109  *               // return STATE.CONTINUE or STATE.ABORT
110  *               return STATE.CONTINUE
111  *          &#125;
112  * <p/>
113  *          &#64;Override
114  *          public String onCompleted() throws Exception &#123;
115  *               // Will be invoked once the response has been fully read or a ResponseComplete exception
116  *               // has been thrown.
117  *               return builder.toString();
118  *          &#125;
119  * <p/>
120  *          &#64;Override
121  *          public void onThrowable(Throwable t) &#123;
122  *          &#125;
123  *      &#125;);
124  * <p/>
125  *      String bodyResponse = f.get();
126  * </pre></blockquote
127  * From any {@link HttpContent} sub classes, you can asynchronously process the response status,headers and body and decide when to
128  * stop the processing the response by throwing a new {link ResponseComplete} at any moment.
129  * <p/>
130  * This class can also be used without the need of {@link AsyncHandler}</p>
131  * <blockquote><pre>
132  *      AsyncHttpClient c = new AsyncHttpClient();
133  *      Future<Response> f = c.prepareGet(TARGET_URL).execute();
134  *      Response r = f.get();
135  * </pre></blockquote>
136  * <p/>
137  * Finally, you can configure the AsyncHttpClient using an {@link AsyncHttpClientConfig} instance</p>
138  * <blockquote><pre>
139  *      AsyncHttpClient c = new AsyncHttpClient(new AsyncHttpClientConfig.Builder().setRequestTimeoutInMs(...).build());
140  *      Future<Response> f = c.prepareGet(TARGET_URL).execute();
141  *      Response r = f.get();
142  * </pre></blockquote>
143  * <p/>
144  * An instance of this class will cache every HTTP 1.1 connections and close them when the {@link AsyncHttpClientConfig#getIdleConnectionTimeoutInMs()}
145  * expires. This object can hold many persistent connections to different host.
146  */

351      * Asynchronous close the {@link AsyncHttpProvider} by spawning a thread and avoid blocking.
352      */
353     public void closeAsynchronously() {
354         final ExecutorService e = Executors.newSingleThreadExecutor();
355         e.submit(new Runnable() {
356             public void run() {
357                 try {
358                     close();
359                 } catch (Throwable t) {
360                     logger.warn("", t);
361                 } finally {
362                     e.shutdown();
363                 }
364             }
365         });
366     }

485     /**
486      * Execute an HTTP request.
487      *
488      * @param request {@link Request}
489      * @param handler an instance of {@link AsyncHandler}
490      * @param <T>     Type of the value that will be returned by the associated {@link java.util.concurrent.Future}
491      * @return a {@link Future} of type T
492      * @throws IOException
493      */
494     public <T> ListenableFuture<T> executeRequest(Request request, AsyncHandler<T> handler) throws IOException {
495 
496         FilterContext fc = new FilterContext.FilterContextBuilder().asyncHandler(handler).request(request).build();
497         fc = preProcessRequest(fc);
498 
499         return httpProvider.execute(fc.getRequest(), fc.getAsyncHandler());
500     }

502     /**
503      * Execute an HTTP request.
504      *
505      * @param request {@link Request}
506      * @return a {@link Future} of type Response
507      * @throws IOException
508      */
509     public ListenableFuture<Response> executeRequest(Request request) throws IOException {
510         FilterContext fc = new FilterContext.FilterContextBuilder().asyncHandler(new AsyncCompletionHandlerBase()).request(request).build();
511         fc = preProcessRequest(fc);
512         return httpProvider.execute(fc.getRequest(), fc.getAsyncHandler());
513     }

515     /**
516      * Configure and execute the associated {@link RequestFilter}. This class may decorate the {@link Request} and {@link AsyncHandler}
517      *
518      * @param fc {@link FilterContext}
519      * @return {@link FilterContext}
520      */
521     private FilterContext preProcessRequest(FilterContext fc) throws IOException {
522         for (RequestFilter asyncFilter : config.getRequestFilters()) {
523             try {
524                 fc = asyncFilter.filter(fc);
525                 if (fc == null) {
526                     throw new NullPointerException("FilterContext is null");
527                 }
528             } catch (FilterException e) {
529                 IOException ex = new IOException();
530                 ex.initCause(e);
531                 throw ex;
532             }
533         }
534 
535         Request request = fc.getRequest();
536         if (fc.getAsyncHandler() instanceof ResumableAsyncHandler) {
537             request = ResumableAsyncHandler.class.cast(fc.getAsyncHandler()).adjustRequestRange(request);
538         }
539 
540         if (request.getRangeOffset() != 0) {
541             RequestBuilder builder = new RequestBuilder(request);
542             builder.setHeader("Range", "bytes=" + request.getRangeOffset() + "-");
543             request = builder.build();
544         }
545         fc = new FilterContext.FilterContextBuilder(fc).request(request).build();
546         return fc;
547     }

549     @SuppressWarnings("unchecked")
550     private final static AsyncHttpProvider loadDefaultProvider(String className, AsyncHttpClientConfig config) {
551         try {
552             Class<AsyncHttpProvider> providerClass = (Class<AsyncHttpProvider>) Thread.currentThread()
553                     .getContextClassLoader().loadClass(className);
554             return providerClass.getDeclaredConstructor(
555                     new Class[]{AsyncHttpClientConfig.class}).newInstance(new Object[]{config});
556         } catch (Throwable t) {
557 
558             if (t instanceof InvocationTargetException) {
559                 final InvocationTargetException ite = (InvocationTargetException) t;
560                 if (logger.isErrorEnabled()) {
561                     logger.error(
562                             "Unable to instantiate provider {}.  Trying other providers.",
563                             className);
564                     logger.error(ite.getCause().toString(), ite.getCause());
565                 }
566             }
567         	
568             // Let's try with another classloader
569             try {
570                 Class<AsyncHttpProvider> providerClass = (Class<AsyncHttpProvider>)
571                         AsyncHttpClient.class.getClassLoader().loadClass(className);
572                 return providerClass.getDeclaredConstructor(
573                         new Class[]{AsyncHttpClientConfig.class}).newInstance(new Object[]{config});
574             } catch (Throwable t2) {
575             }
576 
577             if (logger.isDebugEnabled()) {
578                 logger.debug("Default provider not found {}. Using the {}", DEFAULT_PROVIDER,
579                         JDKAsyncHttpProvider.class.getName());
580             }
581             return new JDKAsyncHttpProvider(config);
582         }
583     }
