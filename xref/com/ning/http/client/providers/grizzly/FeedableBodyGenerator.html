43  /**
44   * A Grizzly-specific {@link BodyGenerator} that allows data to be fed to the
45   * connection in blocking or non-blocking fashion via the use of a {@link Feeder}.
46   *
47   * This class provides two {@link Feeder} implementations for rapid prototyping.
48   * First is the {@link SimpleFeeder} which is simply a listener that asynchronous
49   * data transferring has been initiated.  The second is the {@link NonBlockingFeeder}
50   * which allows reading and feeding data in a non-blocking fashion.
51   *
52   * @author The Grizzly Team
53   * @since 1.7.0
54   */
55  public class FeedableBodyGenerator implements BodyGenerator {

97      /**
98       * Configured the maximum number of bytes that may be pending to be written
99       * to the wire.  If not explicitly configured, the connection's current
100      * configuration will be used instead.
101      *
102      * Once all data has been fed, the connection's max pending bytes configuration
103      * will be restored to its original value.
104      *
105      * @param maxPendingBytes maximum number of bytes that may be queued to
106      *                        be written to the wire.
107      *
108      * @throws IllegalStateException if called after {@link #initializeAsynchronousTransfer(FilterChainContext, HttpRequestPacket)}
109      *  has been called by the {@link GrizzlyAsyncHttpProvider}.
110      * @throws IllegalArgumentException if maxPendingBytes is less than zero and is
111      *  not {@link #UNBOUND} or {@link #DEFAULT}.
112      */
113     @SuppressWarnings("UnusedDeclaration")
114     public synchronized void setMaxPendingBytes(final int maxPendingBytes) {
115         if (maxPendingBytes < DEFAULT) {
116             throw new IllegalArgumentException("Invalid maxPendingBytes value: " + maxPendingBytes);
117         }
118         if (asyncTransferInitiated) {
119             throw new IllegalStateException("Unable to set max pending bytes after async data transfer has been initiated.");
120         }
121         configuredMaxPendingBytes = maxPendingBytes;
122     }

125     /**
126      * Add a {@link Feeder} implementation that will be invoked when writing
127      * without blocking is possible.  This method must be set before dispatching
128      * the request this feeder is associated with.
129      *
130      * @param feeder the {@link Feeder} responsible for providing data.
131      *
132      * @throws IllegalStateException if called after {@link #initializeAsynchronousTransfer(FilterChainContext, HttpRequestPacket)}
133      *  has been called by the {@link GrizzlyAsyncHttpProvider}.
134      * @throws IllegalArgumentException if <code>feeder</code> is <code>null</code>
135      */
136     @SuppressWarnings("UnusedDeclaration")
137     public synchronized void setFeeder(final Feeder feeder) {
138         if (asyncTransferInitiated) {
139             throw new IllegalStateException("Unable to set Feeder after async data transfer has been initiated.");
140         }
141         if (feeder == null) {
142             throw new IllegalArgumentException("Feeder argument cannot be null.");
143         }
144         this.feeder = feeder;
145     }

151     synchronized void initializeAsynchronousTransfer(final FilterChainContext context,
152                                                      final HttpRequestPacket requestPacket)
153     throws IOException {
154 
155         if (asyncTransferInitiated) {
156             throw new IllegalStateException("Async transfer has already been initiated.");
157         }
158         if (feeder == null) {
159             throw new IllegalStateException("No feeder available to perform the transfer.");
160         }
161         assert (context != null);
162         assert (requestPacket != null);
163 
164         this.requestPacket = requestPacket;
165         this.contentBuilder = HttpContent.builder(requestPacket);
166         final Connection c = context.getConnection();
167         origMaxPendingBytes = c.getMaxAsyncWriteQueueSize();
168         if (configuredMaxPendingBytes != DEFAULT) {
169             c.setMaxAsyncWriteQueueSize(configuredMaxPendingBytes);
170         }
171         this.context = context;
172         asyncTransferInitiated = true;
173         final Runnable r = new Runnable() {
174             @Override
175             public void run() {
176                 try {
177                     if (requestPacket.isSecure() &&
178                             (getSSLEngine(context.getConnection()) == null)) {
179                         flushOnSSLHandshakeComplete();
180                     } else {
181                         feeder.flush();
182                     }
183                 } catch (IOException ioe) {
184                     GrizzlyAsyncHttpProvider.HttpTransactionContext ctx =
185                             GrizzlyAsyncHttpProvider.getHttpTransactionContext(
186                                     c);
187                     ctx.abort(ioe);
188                 }
189             }
190         };
191 
192         // If the current thread is a selector thread, we need to execute
193         // the remainder of the task on the worker thread to prevent
194         // it from being blocked.
195         if (isServiceThread()) {
196             c.getTransport().getWorkerThreadPool().execute(r);
197         } else {
198             r.run();
199         }
200     }

211     private void flushOnSSLHandshakeComplete() throws IOException {
212         final FilterChain filterChain = context.getFilterChain();
213         final int idx = filterChain.indexOfType(SSLFilter.class);
214         assert (idx != -1);
215         final SSLFilter filter = (SSLFilter) filterChain.get(idx);
216         final Connection c = context.getConnection();
217         filter.addHandshakeListener(new SSLBaseFilter.HandshakeListener() {
218             public void onStart(Connection connection) {
219             }
220 
221             public void onComplete(Connection connection) {
222                 if (c.equals(connection)) {
223                     filter.removeHandshakeListener(this);
224                     try {
225                         feeder.flush();
226                     } catch (IOException ioe) {
227                         GrizzlyAsyncHttpProvider.HttpTransactionContext ctx =
228                                 GrizzlyAsyncHttpProvider.getHttpTransactionContext(c);
229                         ctx.abort(ioe);
230                     }
231                 }
232             }
233         });
234         filter.handshake(context.getConnection(),  null);
235     }

267     /**
268      * Specifies the functionality all Feeders must implement.  Typically,
269      * developers need not worry about implementing this interface directly.
270      * It should be sufficient, for most use-cases, to simply use the {@link NonBlockingFeeder}
271      * or {@link SimpleFeeder} implementations.
272      */
273     public interface Feeder {
274 
275         /**
276          * This method will be invoked when it's possible to begin feeding
277          * data downstream.  Implementations of this method must use {@link #feed(Buffer, boolean)}
278          * to perform the actual write.
279          *
280          * @throws IOException if an I/O error occurs.
281          */
282         void flush() throws IOException;
283 
284         /**
285          * This method will write the specified {@link Buffer} to the connection.
286          * Be aware that this method may block depending if data is being fed
287          * faster than it can write.  How much data may be queued is dictated
288          * by {@link #setMaxPendingBytes(int)}.  Once this threshold is exceeded,
289          * the method will block until the write queue length drops below the
290          * aforementioned threshold.
291          *
292          * @param buffer the {@link Buffer} to write.
293          * @param last flag indicating if this is the last buffer to send.
294          *
295          * @throws IOException if an I/O error occurs.
296          * @throws java.lang.IllegalArgumentException if <code>buffer</code>
297          *  is <code>null</code>.
298          * @throws java.lang.IllegalStateException if this method is invoked
299          *  before asynchronous transferring has been initiated.
300          *
301          * @see #setMaxPendingBytes(int)
302          */
303         void feed(final Buffer buffer, final boolean last) throws IOException;
304 
305     } // END Feeder

328         /**
329          * {@inheritDoc}
330          */
331         @SuppressWarnings("UnusedDeclaration")
332         public final synchronized void feed(final Buffer buffer, final boolean last)
333         throws IOException {
334             if (buffer == null) {
335                 throw new IllegalArgumentException(
336                         "Buffer argument cannot be null.");
337             }
338             if (!feedableBodyGenerator.asyncTransferInitiated) {
339                 throw new IllegalStateException("Asynchronous transfer has not been initiated.");
340             }
341             blockUntilQueueFree(feedableBodyGenerator.context.getConnection());
342             final HttpContent content =
343                     feedableBodyGenerator.contentBuilder.content(buffer).last(last).build();
344             final CompletionHandler<WriteResult> handler =
345                     ((last) ? new LastPacketCompletionHandler() : null);
346             feedableBodyGenerator.context.write(content, handler);
347         }
348 
349         /**
350          * This method will block if the async write queue is currently larger
351          * than the configured maximum.  The amount of time that this method
352          * will block is dependent on the write timeout of the transport
353          * associated with the specified connection.
354          */
355         private static void blockUntilQueueFree(final Connection c) {
356             if (!c.canWrite()) {
357                 final FutureImpl<Boolean> future =
358                         Futures.createSafeFuture();
359                 // Connection may be obtained by calling FilterChainContext.getConnection().
360                 c.notifyCanWrite(new WriteHandler() {
361 
362                     @Override
363                     public void onWritePossible() throws Exception {
364                         future.result(TRUE);
365                     }
366 
367                     @Override
368                     public void onError(Throwable t) {
369                         future.failure(makeIOException(t));
370                     }
371                 });
372 
373                 block(c, future);
374             }
375         }
376 
377         private static void block(final Connection c,
378                                   final FutureImpl<Boolean> future) {
379             try {
380                 final long writeTimeout =
381                         c.getTransport().getWriteTimeout(MILLISECONDS);
382                 if (writeTimeout != -1) {
383                     future.get(writeTimeout, MILLISECONDS);
384                 } else {
385                     future.get();
386                 }
387             } catch (ExecutionException e) {
388                 GrizzlyAsyncHttpProvider.HttpTransactionContext httpCtx =
389                         getHttpTransactionContext(c);
390                 httpCtx.abort(e.getCause());
391             } catch (Exception e) {
392                 GrizzlyAsyncHttpProvider.HttpTransactionContext httpCtx =
393                         getHttpTransactionContext(c);
394                 httpCtx.abort(e);
395             }
396         }

463     /**
464      * Implementations of this class provide the framework to read data from
465      * some source and feed data to the {@link FeedableBodyGenerator}
466      * without blocking.
467      */
468     @SuppressWarnings("UnusedDeclaration")
469     public static abstract class NonBlockingFeeder extends BaseFeeder {
470 
471 
472         // -------------------------------------------------------- Constructors
473 
474 
475         /**
476          * Constructs the <code>NonBlockingFeeder</code> with the associated
477          * {@link com.ning.http.client.providers.grizzly.FeedableBodyGenerator}.
478          */
479         public NonBlockingFeeder(final FeedableBodyGenerator feedableBodyGenerator) {
480             super(feedableBodyGenerator);
481         }
482 
483 
484         // ------------------------------------------------------ Public Methods
485 
486 
487         /**
488          * Notification that it's possible to send another block of data via
489          * {@link #feed(org.glassfish.grizzly.Buffer, boolean)}.
490          *
491          * It's important to only invoke {@link #feed(Buffer, boolean)}
492          * once per invocation of {@link #canFeed()}.
493          */
494         public abstract void canFeed() throws IOException;
495 
496         /**
497          * @return <code>true</code> if all data has been fed by this feeder,
498          *  otherwise returns <code>false</code>.
499          */
500         public abstract boolean isDone();
501 
502         /**
503          * @return <code>true</code> if data is available to be fed, otherwise
504          *  returns <code>false</code>.  When this method returns <code>false</code>,
505          *  the {@link FeedableBodyGenerator} will call {@link #notifyReadyToFeed(ReadyToFeedListener)}
506          *  by which this {@link NonBlockingFeeder} implementation may signal data is once
507          *  again available to be fed.
508          */
509         public abstract boolean isReady();
510 
511         /**
512          * Callback registration to signal the {@link FeedableBodyGenerator} that
513          * data is available once again to continue feeding.  Once this listener
514          * has been invoked, the NonBlockingFeeder implementation should no longer maintain
515          * a reference to the listener.
516          */
517         public abstract void notifyReadyToFeed(final ReadyToFeedListener listener);
518 
519 
520         // ------------------------------------------------- Methods from Feeder
521 
522 
523         /**
524          * {@inheritDoc}
525          */
526         @Override
527         public synchronized void flush() throws IOException {
528             final Connection c = feedableBodyGenerator.context.getConnection();
529             if (isReady()) {
530                 boolean notReady = writeUntilFullOrDone(c);
531                 if (!isDone()) {
532                     if (notReady) {
533                         notifyReadyToFeed(new ReadyToFeedListenerImpl());
534                     } else {
535                         // write queue is full, leverage WriteListener to let us know
536                         // when it is safe to write again.
537                         c.notifyCanWrite(new WriteHandlerImpl());
538                     }
539                 }
540             } else {
541                 notifyReadyToFeed(new ReadyToFeedListenerImpl());
542             }
543         }
544 
545 
546         // ----------------------------------------------------- Private Methods
547 
548 
549         private boolean writeUntilFullOrDone(final Connection c) throws IOException {
550             while (c.canWrite()) {
551                 if (isReady()) {
552                     canFeed();
553                 } else {
554                     return true;
555                 }
556             }
557             
558             return false;
559         }
