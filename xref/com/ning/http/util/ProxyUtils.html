74      /**
75       * @param config the global config
76       * @param request the request
77       * @return the proxy server to be used for this request (can be null)
78       */
79      public static ProxyServer getProxyServer(AsyncHttpClientConfig config, Request request) {
80          ProxyServer proxyServer = request.getProxyServer();
81          if (proxyServer == null) {
82              ProxyServerSelector selector = config.getProxyServerSelector();
83              if (selector != null) {
84                  proxyServer = selector.select(request.getOriginalURI());
85              }
86          }
87          return ProxyUtils.avoidProxy(proxyServer, request) ? null : proxyServer;
88      }

90      /**
91       * Checks whether proxy should be used according to nonProxyHosts settings of it, or we want to go directly to
92       * target host. If <code>null</code> proxy is passed in, this method returns true -- since there is NO proxy, we
93       * should avoid to use it. Simple hostname pattern matching using "*" are supported, but only as prefixes.
94       * See http://download.oracle.com/javase/1.4.2/docs/guide/net/properties.html
95       *
96       * @param proxyServer
97       * @param request
98       * @return true if we have to avoid proxy use (obeying non-proxy hosts settings), false otherwise.
99       */
100     public static boolean avoidProxy(final ProxyServer proxyServer, final Request request) {
101         return avoidProxy(proxyServer, AsyncHttpProviderUtils.getHost(request.getOriginalURI()));
102     }

104     /**
105      * Checks whether proxy should be used according to nonProxyHosts settings of it, or we want to go directly to
106      * target host. If <code>null</code> proxy is passed in, this method returns true -- since there is NO proxy, we
107      * should avoid to use it. Simple hostname pattern matching using "*" are supported, but only as prefixes.
108      * See http://download.oracle.com/javase/1.4.2/docs/guide/net/properties.html
109      *
110      * @param proxyServer
111      * @param target      the hostname
112      * @return true if we have to avoid proxy use (obeying non-proxy hosts settings), false otherwise.
113      */
114     public static boolean avoidProxy(final ProxyServer proxyServer, final String target) {
115         if (proxyServer != null) {
116             final String targetHost = target.toLowerCase(Locale.ENGLISH);
117 
118             List<String> nonProxyHosts = proxyServer.getNonProxyHosts();
119 
120             if (nonProxyHosts != null) {
121                 for (String nonProxyHost : nonProxyHosts) {
122                     if (nonProxyHost.startsWith("*") && nonProxyHost.length() > 1
123                             && targetHost.endsWith(nonProxyHost.substring(1).toLowerCase(Locale.ENGLISH))) {
124                         return true;
125                     } else if (nonProxyHost.endsWith("*") && nonProxyHost.length() > 1
126                             && targetHost.startsWith(nonProxyHost.substring(0, nonProxyHost.length() - 1).toLowerCase(Locale.ENGLISH))) {
127                         return true;
128                     } else if (nonProxyHost.equalsIgnoreCase(targetHost)) {
129                         return true;
130                     }
131                 }
132             }
133 
134             return false;
135         } else {
136             return true;
137         }
138     }

140     /**
141      * Creates a proxy server instance from the given properties.
142      * <p/>
143      * Currently the default http.* proxy properties are supported as well as properties specific for AHC.
144      *
145      * @param properties the properties to evaluate. Must not be null.
146      * @return a ProxyServer instance or null, if no valid properties were set.
147      * @see <a href="http://download.oracle.com/javase/1.4.2/docs/guide/net/properties.html">Networking Properties</a>
148      * @see #PROXY_HOST
149      * @see #PROXY_PORT
150      * @see #PROXY_PROTOCOL
151      * @see #PROXY_NONPROXYHOSTS
152      */
153     public static ProxyServerSelector createProxyServerSelector(Properties properties) {
154         String host = properties.getProperty(PROXY_HOST);
155 
156         if (host != null) {
157             int port = Integer.valueOf(properties.getProperty(PROXY_PORT, "80"));
158 
159             Protocol protocol;
160             try {
161                 protocol = Protocol.valueOf(properties.getProperty(PROXY_PROTOCOL, "HTTP"));
162             } catch (IllegalArgumentException e) {
163                 protocol = Protocol.HTTP;
164             }
165 
166             ProxyServer proxyServer = new ProxyServer(protocol, host, port, properties.getProperty(PROXY_USER),
167                     properties.getProperty(PROXY_PASSWORD));
168 
169             String nonProxyHosts = properties.getProperty(PROXY_NONPROXYHOSTS);
170             if (nonProxyHosts != null) {
171                 for (String spec : nonProxyHosts.split("\\|")) {
172                     proxyServer.addNonProxyHost(spec);
173                 }
174             }
175 
176             return createProxyServerSelector(proxyServer);
177         }
178 
179         return ProxyServerSelector.NO_PROXY_SELECTOR;
180     }

191     /**
192      * Create a proxy server selector based on the passed in JDK proxy selector.
193      *
194      * @param proxySelector The proxy selector to use.  Must not be null.
195      * @return The proxy server selector.
196      */
197     public static ProxyServerSelector createProxyServerSelector(final ProxySelector proxySelector) {
198         return new ProxyServerSelector() {
199             public ProxyServer select(URI uri) {
200                 List<Proxy> proxies = proxySelector.select(uri);
201                 if (proxies != null) {
202                     // Loop through them until we find one that we know how to use
203                     for (Proxy proxy : proxies) {
204                         switch (proxy.type()) {
205                             case HTTP:
206                                 if (!(proxy.address() instanceof InetSocketAddress)) {
207                                     log.warn("Don't know how to connect to address " + proxy.address());
208                                     return null;
209                                 } else {
210                                     InetSocketAddress address = (InetSocketAddress) proxy.address();
211                                     return new ProxyServer(Protocol.HTTP, address.getHostName(), address.getPort());
212                                 }
213                             case DIRECT:
214                                 return null;
215                             default:
216                                 log.warn("ProxySelector returned proxy type that we don't know how to use: " + proxy.type());
217                                 break;
218                         }
219                     }
220                 }
221                 return null;
222             }
223         };
224     }
