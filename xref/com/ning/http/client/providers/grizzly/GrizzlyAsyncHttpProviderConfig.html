25   * {@link AsyncHttpProviderConfig} implementation that allows customization

34  public class GrizzlyAsyncHttpProviderConfig implements AsyncHttpProviderConfig<GrizzlyAsyncHttpProviderConfig.Property,Object> {

36      /**
37       * Grizzly-specific customization properties.  Each property describes
38       * what it's used for, what the default value is (if any), and what
39       * the expected type the value of the property should be.
40       */
41      public static enum Property {
42  
43          /**
44           * If this property is specified with a custom {@link TransportCustomizer}
45           * instance, the {@link TCPNIOTransport} instance that is created by
46           * {@link GrizzlyAsyncHttpProvider} will be passed to the customizer
47           * bypassing all default configuration of the transport typically performed
48           * by the provider. The type of the value associated with this property
49           * must be <code>TransportCustomizer.class</code>.
50           *
51           * @see TransportCustomizer
52           */
53          TRANSPORT_CUSTOMIZER(TransportCustomizer.class),
54  
55  
56          /**
57           * Defines the maximum HTTP packet header size.
58           */
59          MAX_HTTP_PACKET_HEADER_SIZE(Integer.class, HttpCodecFilter.DEFAULT_MAX_HTTP_PACKET_HEADER_SIZE),
60  
61  
62          /**
63           * By default, Websocket messages that are fragmented will be buffered.  Once all
64           * fragments have been accumulated, the appropriate onMessage() call back will be
65           * invoked with the complete message.  If this functionality is not desired, set
66           * this property to false.
67           */
68          BUFFER_WEBSOCKET_FRAGMENTS(Boolean.class, true)
69  
70          ;
71          
72          
73          final Object defaultValue;
74          final Class<?> type;
75          
76          private Property(final Class<?> type, final Object defaultValue) {
77              this.type = type;
78              this.defaultValue = defaultValue;
79          }
80          
81          private Property(final Class<?> type) {
82              this(type, null);
83          }

99       * @throws IllegalArgumentException if the type of the specified value
100      *  does not match the expected type of the specified {@link Property}.
101      */
102     @Override
103     public AsyncHttpProviderConfig addProperty(Property name, Object value) {
104         if (name == null) {
105             return this;
106         }
107         if (value == null) {
108             if (name.hasDefaultValue()) {
109                 value = name.defaultValue;
110             } else {
111                 return this;
112             }
113         } else {
114             if (!name.type.isAssignableFrom(value.getClass())) {
115                 throw new IllegalArgumentException(
116                         String.format(
117                                 "The value of property [%s] must be of type [%s].  Type of value provided: [%s].",
118                                 name.name(),
119                                 name.type.getName(),
120                                 value.getClass().getName()));
121             }
122         }
123         attributes.put(name, value);
124         return this;
125     }
