13  package com.ning.http.multipart;
14  
15  import java.util.Timer;
16  import java.util.TimerTask;
17  
18  /**
19   * @author Gail Hernandez
20   */
21  public class FilePartStallHandler extends TimerTask {
22  	public FilePartStallHandler(long waitTime, FilePart filePart) {
23  		_waitTime = waitTime;
24  		_failed = false;
25  		_written = false;
26  	}
27  	
28  	public void completed() {
29  		if(_waitTime > 0) {
30  			_timer.cancel();
31  		}
32  	}
33  
34  	public boolean isFailed() {
35  		return _failed;
36  	}
37  
38  	public void run() {
39  		if(!_written) {
40  			_failed = true;
41  			_timer.cancel();
42  		}
43  		_written = false;
44  	}
45  
46  	public void start() {
47  		if(_waitTime > 0) {
48  			_timer = new Timer();
49  			_timer.scheduleAtFixedRate(this, _waitTime, _waitTime);
50  		}
51  	}
52  	
53  	public void writeHappened() {
54  		_written = true;
55  	}
56  
57  	private long _waitTime;
58  	private Timer _timer;
59  	private boolean _failed;
60  	private boolean _written;
61  }
