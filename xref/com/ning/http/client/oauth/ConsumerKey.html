19  /**
20   * Value class for OAuth consumer keys.
21   */
22  public class ConsumerKey {
23      private final String key;
24      private final String secret;
25  
26      public ConsumerKey(String key, String secret) {
27          this.key = key;
28          this.secret = secret;
29      }
30  
31      public String getKey() {
32          return key;
33      }
34  
35      public String getSecret() {
36          return secret;
37      }
38  
39      @Override
40      public String toString() {
41          StringBuilder sb = new StringBuilder("{Consumer key, key=");
42          appendValue(sb, key);
43          sb.append(", secret=");
44          appendValue(sb, secret);
45          sb.append("}");
46          return sb.toString();
47      }
48  
49      private void appendValue(StringBuilder sb, String value) {
50          if (value == null) {
51              sb.append("null");
52          } else {
53              sb.append('"');
54              sb.append(value);
55              sb.append('"');
56          }
57      }
58  
59      @Override
60      public int hashCode() {
61          return key.hashCode() + secret.hashCode();
62      }
63  
64      @Override
65      public boolean equals(Object o) {
66          if (o == this) return true;
67          if (o == null || o.getClass() != getClass()) return false;
68          ConsumerKey other = (ConsumerKey) o;
69          return key.equals(other.key) && secret.equals(other.secret);
70      }
