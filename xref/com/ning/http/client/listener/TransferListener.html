13  package com.ning.http.client.listener;
14  
15  import com.ning.http.client.FluentCaseInsensitiveStringsMap;
16  
17  import java.io.IOException;
18  import java.nio.ByteBuffer;
19  
20  /**
21   * A simple interface an application can implements in order to received byte transfer information.
22   */
23  public interface TransferListener {
24  
25      /**
26       * Invoked when the request bytes are starting to get send.
27       */
28      public void onRequestHeadersSent(FluentCaseInsensitiveStringsMap headers);
29  
30      /**
31       * Invoked when the response bytes are starting to get received.
32       */
33      public void onResponseHeadersReceived(FluentCaseInsensitiveStringsMap headers);
34  
35      /**
36       * Invoked every time response's chunk are received.
37       *
38       * @param buffer a {@link ByteBuffer}
39       */
40      public void onBytesReceived(ByteBuffer buffer) throws IOException;
41  
42      /**
43       * Invoked every time request's chunk are sent.
44       *
45       * @param buffer a {@link ByteBuffer}
46       */
47      public void onBytesSent(ByteBuffer buffer);
48  
49      /**
50       * Invoked when the response bytes are been fully received.
51       */
52      public void onRequestResponseCompleted();
53  
54      /**
55       * Invoked when there is an unexpected issue.
56       *
57       * @param t a {@link Throwable}
58       */
59      public void onThrowable(Throwable t);
60  }
