19  /**
20   * A companion for CookieDecoder that parses key-value pairs (cookie name/value
21   * and attributes).
22   * 
23   * @author slandelle
24   */
25  class KeyValuePairsParser {
26  
27      private final TimeConverter timeBuilder;
28      private String name;
29      private String value;
30      private String rawValue;
31      private String domain;
32      private String path;
33      private long expires = -1L;
34      private int maxAge = -1;
35      private boolean secure;
36      private boolean httpOnly;
37  
38      /**
39       * @param timeBuilder used for parsing expires attribute
40       */
41      public KeyValuePairsParser(TimeConverter timeBuilder) {
42          this.timeBuilder = timeBuilder;
43      }
44  
45      public Cookie cookie() {
46          return name != null ? new Cookie(name, value, rawValue, domain, path, expires, maxAge, secure, httpOnly) : null;
47      }
48  
49      /**
50       * Parse and store a key-value pair. First one is considered to be the
51       * cookie name/value. Unknown attribute names are silently discarded.
52       * 
53       * @param header the HTTP header
54       * @param keyStart where the key starts in the header
55       * @param keyEnd where the key ends in the header
56       * @param value the decoded value
57       * @param rawValue the raw value (only non null for cookie value)
58       */
59      public void parseKeyValuePair(String header, int keyStart, int keyEnd, String value, String rawValue) {
60  
61          if (name == null)
62              setCookieNameValue(header, keyStart, keyEnd, value, rawValue);
63          else
64              setCookieAttribute(header, keyStart, keyEnd, value);
65      }
66  
67      private void setCookieNameValue(String header, int keyStart, int keyEnd, String value, String rawValue) {
68          name = header.substring(keyStart, keyEnd);
69          this.value = value;
70          this.rawValue = rawValue;
71      }
72  
73      private void setCookieAttribute(String header, int keyStart, int keyEnd, String value) {
74  
75          int length = keyEnd - keyStart;
76  
77          if (length == 4)
78              parse4(header, keyStart, value);
79          else if (length == 6)
80              parse6(header, keyStart, value);
81          else if (length == 7)
82              parse7(header, keyStart, value);
83          else if (length == 8)
84              parse8(header, keyStart, value);
85      }

87      private boolean isPath(char c0, char c1, char c2, char c3) {
88          return (c0 == 'P' || c0 == 'p') && //
89                  (c1 == 'a' || c1 == 'A') && //
90                  (c2 == 't' || c2 == 'T') && //
91                  (c3 == 'h' || c3 == 'H');
92      }
93  
94      private void parse4(String header, int nameStart, String value) {
95  
96          char c0 = header.charAt(nameStart);
97          char c1 = header.charAt(nameStart + 1);
98          char c2 = header.charAt(nameStart + 2);
99          char c3 = header.charAt(nameStart + 3);
100 
101         if (isPath(c0, c1, c2, c3))
102             path = value;
103     }
104 
105     private boolean isDomain(char c0, char c1, char c2, char c3, char c4, char c5) {
106         return (c0 == 'D' || c0 == 'd') && //
107                 (c1 == 'o' || c1 == 'O') && //
108                 (c2 == 'm' || c2 == 'M') && //
109                 (c3 == 'a' || c3 == 'A') && //
110                 (c4 == 'i' || c4 == 'I') && //
111                 (c5 == 'n' || c5 == 'N');
112     }
113 
114     private boolean isSecure(char c0, char c1, char c2, char c3, char c4, char c5) {
115         return (c0 == 'S' || c0 == 's') && //
116                 (c1 == 'e' || c1 == 'E') && //
117                 (c2 == 'c' || c2 == 'C') && //
118                 (c3 == 'u' || c3 == 'U') && //
119                 (c4 == 'r' || c4 == 'R') && //
120                 (c5 == 'e' || c5 == 'E');
121     }
122 
123     private void parse6(String header, int nameStart, String value) {
124 
125         char c0 = header.charAt(nameStart);
126         char c1 = header.charAt(nameStart + 1);
127         char c2 = header.charAt(nameStart + 2);
128         char c3 = header.charAt(nameStart + 3);
129         char c4 = header.charAt(nameStart + 4);
130         char c5 = header.charAt(nameStart + 5);
131 
132         if (isDomain(c0, c1, c2, c3, c4, c5))
133             domain = value;
134         else if (isSecure(c0, c1, c2, c3, c4, c5))
135             secure = true;
136     }

138     private boolean isExpires(char c0, char c1, char c2, char c3, char c4, char c5, char c6) {
139         return (c0 == 'E' || c0 == 'e') && //
140                 (c1 == 'x' || c1 == 'X') && //
141                 (c2 == 'p' || c2 == 'P') && //
142                 (c3 == 'i' || c3 == 'I') && //
143                 (c4 == 'r' || c4 == 'R') && //
144                 (c5 == 'e' || c5 == 'E') && //
145                 (c6 == 's' || c6 == 'S');
146     }
147 
148     private boolean isMaxAge(char c0, char c1, char c2, char c3, char c4, char c5, char c6) {
149         return (c0 == 'M' || c0 == 'm') && //
150                 (c1 == 'a' || c1 == 'A') && //
151                 (c2 == 'x' || c2 == 'X') && //
152                 (c3 == '-') && //
153                 (c4 == 'A' || c4 == 'a') && //
154                 (c5 == 'g' || c5 == 'G') && //
155                 (c6 == 'e' || c6 == 'E');
156     }
157 
158     private void setExpire(String value) {
159 
160         RFC2616Date dateElements = new RFC2616DateParser(value).parse();
161         if (dateElements != null) {
162             try {
163                 expires = timeBuilder.toTime(dateElements);
164             } catch (Exception e1) {
165                 // ignore failure to parse -> treat as session cookie
166             }
167         }
168     }
169 
170     private void setMaxAge(String value) {
171         try {
172             maxAge = Math.max(Integer.valueOf(value), 0);
173         } catch (NumberFormatException e1) {
174             // ignore failure to parse -> treat as session cookie
175         }
176     }
177 
178     private void parse7(String header, int nameStart, String value) {
179 
180         char c0 = header.charAt(nameStart);
181         char c1 = header.charAt(nameStart + 1);
182         char c2 = header.charAt(nameStart + 2);
183         char c3 = header.charAt(nameStart + 3);
184         char c4 = header.charAt(nameStart + 4);
185         char c5 = header.charAt(nameStart + 5);
186         char c6 = header.charAt(nameStart + 6);
187 
188         if (isExpires(c0, c1, c2, c3, c4, c5, c6))
189             setExpire(value);
190 
191         else if (isMaxAge(c0, c1, c2, c3, c4, c5, c6))
192             setMaxAge(value);
193     }

195     private boolean isHttpOnly(char c0, char c1, char c2, char c3, char c4, char c5, char c6, char c7) {
196         return (c0 == 'H' || c0 == 'h') && //
197                 (c1 == 't' || c1 == 'T') && //
198                 (c2 == 't' || c2 == 'T') && //
199                 (c3 == 'p' || c3 == 'P') && //
200                 (c4 == 'O' || c4 == 'o') && //
201                 (c5 == 'n' || c5 == 'N') && //
202                 (c6 == 'l' || c6 == 'L') && //
203                 (c7 == 'y' || c7 == 'Y');
204     }

206     private void parse8(String header, int nameStart, String value) {
207 
208         char c0 = header.charAt(nameStart);
209         char c1 = header.charAt(nameStart + 1);
210         char c2 = header.charAt(nameStart + 2);
211         char c3 = header.charAt(nameStart + 3);
212         char c4 = header.charAt(nameStart + 4);
213         char c5 = header.charAt(nameStart + 5);
214         char c6 = header.charAt(nameStart + 6);
215         char c7 = header.charAt(nameStart + 7);
216 
217         if (isHttpOnly(c0, c1, c2, c3, c4, c5, c6, c7))
218             httpOnly = true;
219     }
